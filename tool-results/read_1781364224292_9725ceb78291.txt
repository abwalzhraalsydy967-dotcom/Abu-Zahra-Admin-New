   307→   307→def _get_real_ip(request):
   308→   308→    """Extract real client IP from request, handling reverse proxy headers."""
   309→   309→    forwarded = request.headers.get("X-Forwarded-For", "").strip()
   310→   310→    if forwarded:
   311→   311→        return forwarded.split(",")[0].strip()
   312→   312→    real_ip = request.headers.get("X-Real-IP", "").strip()
   313→   313→    if real_ip:
   314→   314→        return real_ip
   315→   315→    return request.remote or ""

   402→   402→def format_uptime(seconds):
   403→   403→    d = seconds // 86400
   404→   404→    h = (seconds % 86400) // 3600
   405→   405→    m = (seconds % 3600) // 60
   406→   406→    s = seconds % 60
   407→   407→    parts = []
   408→   408→    if d: parts.append(f"{d}d")
   409→   409→    if h: parts.append(f"{h}h")
   410→   410→    if m: parts.append(f"{m}m")
   411→   411→    parts.append(f"{s}s")
   412→   412→    return " ".join(parts)
