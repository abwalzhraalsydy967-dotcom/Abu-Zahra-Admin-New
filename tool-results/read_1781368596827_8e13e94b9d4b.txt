    32→@web.middleware
    33→async def cors_middleware(request, handler):
    34→    allowed_origins = ['https://alsydyabwalzhra.online', 'http://localhost:8443']
    35→    origin = request.headers.get('Origin', '')
    36→    if request.method == 'OPTIONS':
    37→        response = web.Response(status=204)
    38→        if origin in allowed_origins:
    39→            response.headers['Access-Control-Allow-Origin'] = origin
    40→            response.headers['Access-Control-Allow-Methods'] = 'GET, POST, PUT, DELETE, OPTIONS'
    41→            response.headers['Access-Control-Allow-Headers'] = 'Authorization, Content-Type, X-Device-Token'
    42→            response.headers['Access-Control-Allow-Credentials'] = 'true'
    43→        return response
    44→    response = await handler(request)
    45→    if origin in allowed_origins:
    46→        response.headers['Access-Control-Allow-Origin'] = origin

    48→        response.headers['Access-Control-Allow-Headers'] = 'Authorization, Content-Type, X-Device-Token'
    49→        response.headers['Access-Control-Allow-Credentials'] = 'true'
    50→    return response

   331→def _get_real_ip(request):
   332→    """Extract real client IP from request, handling reverse proxy headers."""
   333→    forwarded = request.headers.get("X-Forwarded-For", "").strip()
   334→    if forwarded:
   335→        return forwarded.split(",")[0].strip()
   336→    real_ip = request.headers.get("X-Real-IP", "").strip()
   337→    if real_ip:
   338→        return real_ip
   339→    return request.remote or ""

   426→def format_uptime(seconds):
   427→    d = seconds // 86400
   428→    h = (seconds % 86400) // 3600
   429→    m = (seconds % 3600) // 60
   430→    s = seconds % 60
   431→    parts = []
   432→    if d: parts.append(f"{d}d")
   433→    if h: parts.append(f"{h}h")
   434→    if m: parts.append(f"{m}m")
   435→    parts.append(f"{s}s")
   436→    return " ".join(parts)

   513→    commands = load_json(COMMANDS_FILE, [])

   519→        "status": "pending",
   520→        "created_at": ts(),
   521→        "sent_at": None,
   522→        "result": None,
   523→    }
   524→    commands.append(cmd)
   525→    if len(commands) > 1000:
   526→        commands = commands[-1000:]
   527→    save_json(COMMANDS_FILE, commands)
   528→    append_event("Command queued", {"device_id": device_id, "command": command, "cmd_id": cmd["id"]})
   529→    firebase_push_command(cmd)
   530→    return cmd

   553→def create_session(username, password, ip="", ua=""):
   554→    settings = load_settings()
   555→    if password != settings.get("admin_password", "admin"):
   556→        return None
   557→    sessions = load_json(SESSIONS_FILE, [])
   558→    token = secrets.token_urlsafe(32)
   559→    session = {
   560→        "token": token,
   561→        "username": username,
   562→        "created_at": datetime.now(timezone.utc).isoformat(),
   563→        "expires_at": (datetime.now(timezone.utc) + timedelta(hours=24)).isoformat(),
   564→        "ip": ip,
   565→        "user_agent": ua,
   566→    }
   567→    sessions.append(session)
   568→    if len(sessions) > 100:
   569→        sessions = sessions[-100:]
   570→    save_json(SESSIONS_FILE, sessions)
   571→    append_event("Web login", {"username": username, "ip": ip})
   572→    return session

   575→def validate_session(token):
   576→    sessions = load_json(SESSIONS_FILE, [])
   577→    now = datetime.now(timezone.utc)
   578→    for s in sessions:
   579→        if s.get("token") == token:
   580→            try:
   581→                expires = datetime.fromisoformat(s.get("expires_at", "")).replace(tzinfo=timezone.utc)
   582→                if now > expires:
   583→                    return None
   584→            except Exception:
   585→                return None
   586→            return s
   587→    return None

   590→def delete_session(token):
   591→    sessions = load_json(SESSIONS_FILE, [])
   592→    new_sessions = [s for s in sessions if s.get("token") != token]
   593→    save_json(SESSIONS_FILE, new_sessions)

   622→async def firebase_get(path):
   623→    """GET data from Firebase RTDB.
   624→    يعمل بدون مصادقة إذا كانت القواعد تسمح بالوصول العام.
   625→    أو مع Database Secret إذا تم تعيين FIREBASE_DB_SECRET.

   627→    try:

   629→        if FIREBASE_DB_SECRET:
   630→            url += f"?auth={FIREBASE_DB_SECRET}"

   644→async def firebase_set(path, data):
   645→    """SET data in Firebase RTDB - with optional Database Secret auth.
   646→    If data is None, uses DELETE instead of PUT (to properly remove the key).
   647→    Gracefully returns False if Firebase is unavailable."""
   648→    try:
   649→        url = f"{FIREBASE_RTDB_URL}/{path}.json"
   650→        if FIREBASE_DB_SECRET:
   651→            url += f"?auth={FIREBASE_DB_SECRET}"
   652→        session = get_tg_session()
   653→        if data is None:
   654→            # حذف المسار بدلاً من تعيينه إلى null
   655→            async with session.delete(url, timeout=aiohttp.ClientTimeout(total=10)) as resp:
   656→                ok = resp.status in (200, 204)
   657→                if not ok:
   658→                    body = await resp.text()
   659→                    log.warning("Firebase DELETE %s failed: status=%d body=%s", path, resp.status, body[:200])
   660→                return ok
   661→        else:
   662→            async with session.put(url, json=data, timeout=aiohttp.ClientTimeout(total=10)) as resp:
   663→                ok = resp.status in (200, 204)
   664→                if ok:
   665→                    log.debug("Firebase SET %s OK", path)
   666→                else:
   667→                    body = await resp.text()
   668→                    log.warning("Firebase SET %s failed: status=%d body=%s", path, resp.status, body[:200])
   669→                return ok
   670→    except Exception as exc:
   671→        log.error("Firebase SET %s failed: %s", path, exc)
   672→        return False

   706→async def _firebase_push_cmd_async(cmd):
   707→    """Async: Push command to Firebase /commands/{device_id}/{cmd_id}"""
   708→    device_id = cmd.get("device_id", "")
   709→    cmd_id = cmd.get("id", "")
   710→    if not device_id or not cmd_id:
   711→        return
   712→    try:
   713→        ok = await firebase_set(f"commands/{device_id}/{cmd_id}", {
   714→            "id": cmd["id"],
   715→            "device_id": cmd["device_id"],
   716→            "command": cmd["command"],
   717→            "params": cmd.get("params", {}),
   718→            "status": "pending",
   719→            "created_at": cmd["created_at"],
   720→            "server_domain": SERVER_DOMAIN,
   721→            "server_port": SERVER_PORT,
   722→        })
   723→        if ok:
   724→            log.info("Firebase: Command %s pushed for device %s", cmd_id, device_id)
   725→            # Schedule deletion after 60 seconds (app polls every 10s, gives plenty of time)
   726→            async def _delayed_delete():
   727→                await asyncio.sleep(60)
   728→                try:
   729→                    await firebase_set(f"commands/{device_id}/{cmd_id}", None)
   730→                    log.info("Firebase: Command %s auto-deleted after timeout", cmd_id)
   731→                except Exception:
   732→                    pass
   733→            asyncio.ensure_future(_delayed_delete())
   734→        else:
   735→            log.warning("Firebase: Failed to push command %s", cmd_id)
   736→    except Exception as exc:
   737→        log.error("Firebase push command error: %s", exc)

   768→async def verify_link_code(code):
   769→    """Verify link code - checks Firebase first, then local fallback."""
   770→    # 1. التحقق من Firebase
   771→    fb_data = await firebase_get(f"link_codes/{code}")
   772→    if fb_data is not None:
   773→        if fb_data.get("used"):
   774→            return {"ok": False, "error": "Code already used"}
   775→        return {"ok": True, "code_entry": fb_data}
   776→    # 2. التحقق من الملف المحلي
   777→    codes = load_json(LINK_CODES_FILE, [])
   778→    for entry in codes:
   779→        if entry.get("code") == code:
   780→            if entry.get("used"):
   781→                return {"ok": False, "error": "Code already used"}
   782→            return {"ok": True, "code_entry": entry}
   783→    return {"ok": False, "error": "Invalid code"}

   787→    """Mark link code as used in Firebase + local."""
   788→    now = datetime.now(timezone.utc).isoformat()
   789→    # 1. تحديث Firebase
   790→    await firebase_update(f"link_codes/{code}", {
   791→        "used": True,
   792→        "device_id": device_id,
   793→        "used_at": now,
   794→    })
   795→    # 2. تحديث محلي
   796→    codes = load_json(LINK_CODES_FILE, [])
   797→    for entry in codes:
   798→        if entry.get("code") == code:
   799→            entry["used"] = True
   800→            entry["device_id"] = device_id
   801→            entry["used_at"] = now
   802→            save_json(LINK_CODES_FILE, codes)
   803→            return True
   804→    return False

   831→async def send_message(chat_id, text, parse_mode="HTML", reply_markup=None, disable_notification=False):
   832→    global messages_sent
   833→    # === ANTI-SPAM: Deduplication + Rate Limiting ===
   834→    now = time.time()
   835→    chat_key = str(chat_id)
   836→    text_hash = hashlib.md5(text[:200].encode()).hexdigest() if text else ""
   837→
   838→    # Check: same message to same chat within 60 seconds?
   839→    dedup_key = f"{chat_key}:{text_hash}"
   840→    last_sent = _message_dedup.get(dedup_key, 0)
   841→    if now - last_sent < 10 and text_hash:
   842→        log.warning("DEDUP BLOCKED: chat=%s hash=%s", chat_key, text_hash[:8])
   843→        return {"ok": False, "description": "dedup_blocked"}
   844→
   845→    # Check: more than 5 messages to same chat within 30 seconds?
   846→    recent = _chat_rate_counter.get(chat_key, [])
   847→    recent = [t for t in recent if now - t < 30]
   848→    if len(recent) >= 10:
   849→        log.warning("RATE BLOCKED: chat=%s count=%d", chat_key, len(recent))
   850→        return {"ok": False, "description": "rate_limited"}
   851→    recent.append(now)
   852→    _chat_rate_counter[chat_key] = recent
   853→
   854→    # Register in dedup table
   855→    _message_dedup[dedup_key] = now
   856→    expired = [k for k, v in _message_dedup.items() if now - v > 30]
   857→    for k in expired:
   858→        del _message_dedup[k]
   859→
   860→    payload = {"chat_id": chat_id, "text": text, "parse_mode": parse_mode}
   861→    if reply_markup:
   862→        payload["reply_markup"] = reply_markup
   863→    if disable_notification:
   864→        payload["disable_notification"] = True
   865→    result = await tg_request("sendMessage", payload)
   866→    if result and result.get("ok"):
   867→        messages_sent += 1
   868→    return result
