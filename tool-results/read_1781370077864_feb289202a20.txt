    32→@web.middleware
    33→async def cors_middleware(request, handler):
    34→    allowed_origins = ['https://alsydyabwalzhra.online', 'http://localhost:8443']
    35→    origin = request.headers.get('Origin', '')
    36→    if request.method == 'OPTIONS':
    37→        response = web.Response(status=204)
    38→        if origin in allowed_origins:
    39→            response.headers['Access-Control-Allow-Origin'] = origin
    40→            response.headers['Access-Control-Allow-Methods'] = 'GET, POST, PUT, DELETE, OPTIONS'
    41→            response.headers['Access-Control-Allow-Headers'] = 'Authorization, Content-Type, X-Device-Token'
    42→            response.headers['Access-Control-Allow-Credentials'] = 'true'
    43→        return response
    44→    response = await handler(request)
    45→    if origin in allowed_origins:
    46→        response.headers['Access-Control-Allow-Origin'] = origin
    47→        response.headers['Access-Control-Allow-Methods'] = 'GET, POST, PUT, DELETE, OPTIONS'
    48→        response.headers['Access-Control-Allow-Headers'] = 'Authorization, Content-Type, X-Device-Token'
    49→        response.headers['Access-Control-Allow-Credentials'] = 'true'
    50→    return response

   342→def _get_real_ip(request):
   343→    """Extract real client IP from request, handling reverse proxy headers."""
   344→    forwarded = request.headers.get("X-Forwarded-For", "").strip()
   345→    if forwarded:
   346→        return forwarded.split(",")[0].strip()
   347→    real_ip = request.headers.get("X-Real-IP", "").strip()
   348→    if real_ip:
   349→        return real_ip
   350→    return request.remote or ""

   437→def format_uptime(seconds):
   438→    d = seconds // 86400
   439→    h = (seconds % 86400) // 3600
   440→    m = (seconds % 3600) // 60
   441→    s = seconds % 60
   442→    parts = []
   443→    if d: parts.append(f"{d}d")
   444→    if h: parts.append(f"{h}h")
   445→    if m: parts.append(f"{m}m")
   446→    parts.append(f"{s}s")
   447→    return " ".join(parts)

   524→    commands = load_json(COMMANDS_FILE, [])

   531→        "created_at": ts(),

   534→    }

   537→        commands = commands[-1000:]
   538→    save_json(COMMANDS_FILE, commands)
   539→    append_event("Command queued", {"device_id": device_id, "command": command, "cmd_id": cmd["id"]})
   540→    firebase_push_command(cmd)
   541→    return cmd

   564→def create_session(username, password, ip="", ua=""):
   565→    settings = load_settings()
   566→    if password != settings.get("admin_password", "admin"):
   567→        return None
   568→    sessions = load_json(SESSIONS_FILE, [])
   569→    token = secrets.token_urlsafe(32)
   570→    session = {
   571→        "token": token,
   572→        "username": username,
   573→        "created_at": datetime.now(timezone.utc).isoformat(),
   574→        "expires_at": (datetime.now(timezone.utc) + timedelta(hours=24)).isoformat(),
   575→        "ip": ip,
   576→        "user_agent": ua,
   577→    }
   578→    sessions.append(session)
   579→    if len(sessions) > 100:
   580→        sessions = sessions[-100:]
   581→    save_json(SESSIONS_FILE, sessions)
   582→    append_event("Web login", {"username": username, "ip": ip})
   583→    return session

   655→async def firebase_set(path, data):
   656→    """SET data in Firebase RTDB - with optional Database Secret auth.
   657→    If data is None, uses DELETE instead of PUT (to properly remove the key).
   658→    Gracefully returns False if Firebase is unavailable."""
   659→    try:
   660→        url = f"{FIREBASE_RTDB_URL}/{path}.json"
   661→        if FIREBASE_DB_SECRET:
   662→            url += f"?auth={FIREBASE_DB_SECRET}"
   663→        session = get_tg_session()
   664→        if data is None:
   665→            # حذف المسار بدلاً من تعيينه إلى null
   666→            async with session.delete(url, timeout=aiohttp.ClientTimeout(total=10)) as resp:
   667→                ok = resp.status in (200, 204)
   668→                if not ok:
   669→                    body = await resp.text()
   670→                    log.warning("Firebase DELETE %s failed: status=%d body=%s", path, resp.status, body[:200])
   671→                return ok
   672→        else:
   673→            async with session.put(url, json=data, timeout=aiohttp.ClientTimeout(total=10)) as resp:
   674→                ok = resp.status in (200, 204)
   675→                if ok:
   676→                    log.debug("Firebase SET %s OK", path)
   677→                else:
   678→                    body = await resp.text()
   679→                    log.warning("Firebase SET %s failed: status=%d body=%s", path, resp.status, body[:200])
   680→                return ok
   681→    except Exception as exc:
   682→        log.error("Firebase SET %s failed: %s", path, exc)
   683→        return False

   718→    """Async: Push command to Firebase /commands/{device_id}/{cmd_id}"""
   719→    device_id = cmd.get("device_id", "")
   720→    cmd_id = cmd.get("id", "")
   721→    if not device_id or not cmd_id:
   722→        return
   723→    try:
   724→        ok = await firebase_set(f"commands/{device_id}/{cmd_id}", {
   725→            "id": cmd["id"],
   726→            "device_id": cmd["device_id"],
   727→            "command": cmd["command"],
   728→            "params": cmd.get("params", {}),
   729→            "status": "pending",
   730→            "created_at": cmd["created_at"],
   731→            "server_domain": SERVER_DOMAIN,
   732→            "server_port": SERVER_PORT,
   733→        })
   734→        if ok:
   735→            log.info("Firebase: Command %s pushed for device %s", cmd_id, device_id)
   736→            # Schedule deletion after 60 seconds (app polls every 10s, gives plenty of time)
   737→            async def _delayed_delete():
   738→                await asyncio.sleep(60)
   739→                try:
   740→                    await firebase_set(f"commands/{device_id}/{cmd_id}", None)
   741→                    log.info("Firebase: Command %s auto-deleted after timeout", cmd_id)
   742→                except Exception:
   743→                    pass
   744→            asyncio.ensure_future(_delayed_delete())
   745→        else:
   746→            log.warning("Firebase: Failed to push command %s", cmd_id)
   747→    except Exception as exc:
   748→        log.error("Firebase push command error: %s", exc)

   779→async def verify_link_code(code):
   780→    """Verify link code - checks Firebase first, then local fallback."""
   781→    # 1. التحقق من Firebase
   782→    fb_data = await firebase_get(f"link_codes/{code}")
   783→    if fb_data is not None:
   784→        if fb_data.get("used"):
   785→            return {"ok": False, "error": "Code already used"}
   786→        return {"ok": True, "code_entry": fb_data}
   787→    # 2. التحقق من الملف المحلي
   788→    codes = load_json(LINK_CODES_FILE, [])
   789→    for entry in codes:
   790→        if entry.get("code") == code:
   791→            if entry.get("used"):
   792→                return {"ok": False, "error": "Code already used"}
   793→            return {"ok": True, "code_entry": entry}
   794→    return {"ok": False, "error": "Invalid code"}

   842→async def send_message(chat_id, text, parse_mode="HTML", reply_markup=None, disable_notification=False):
   843→    global messages_sent
   844→    # === ANTI-SPAM: Deduplication + Rate Limiting ===
   845→    now = time.time()
   846→    chat_key = str(chat_id)
   847→    text_hash = hashlib.md5(text[:200].encode()).hexdigest() if text else ""
   848→
   849→    # Check: same message to same chat within 60 seconds?
   850→    dedup_key = f"{chat_key}:{text_hash}"
   851→    last_sent = _message_dedup.get(dedup_key, 0)
   852→    if now - last_sent < 10 and text_hash:
   853→        log.warning("DEDUP BLOCKED: chat=%s hash=%s", chat_key, text_hash[:8])
   854→        return {"ok": False, "description": "dedup_blocked"}
   855→
   856→    # Check: more than 5 messages to same chat within 30 seconds?
   857→    recent = _chat_rate_counter.get(chat_key, [])
   858→    recent = [t for t in recent if now - t < 30]
   859→    if len(recent) >= 10:
   860→        log.warning("RATE BLOCKED: chat=%s count=%d", chat_key, len(recent))
   861→        return {"ok": False, "description": "rate_limited"}
   862→    recent.append(now)
   863→    _chat_rate_counter[chat_key] = recent
   864→
   865→    # Register in dedup table
   866→    _message_dedup[dedup_key] = now
   867→    expired = [k for k, v in _message_dedup.items() if now - v > 30]
   868→    for k in expired:
   869→        del _message_dedup[k]
   870→
   871→    payload = {"chat_id": chat_id, "text": text, "parse_mode": parse_mode}
   872→    if reply_markup:
   873→        payload["reply_markup"] = reply_markup
   874→    if disable_notification:
   875→        payload["disable_notification"] = True
   876→    result = await tg_request("sendMessage", payload)
   877→    if result and result.get("ok"):
   878→        messages_sent += 1
   879→    return result
