   324→    fun enableBiometric(context: Context): String {
   325→        return try {
   326→            // Check if biometric is available
   327→            val biometricManager = BiometricManager.from(context)
   328→            val canAuthenticate = when (biometricManager.canAuthenticate(BiometricManager.Authenticators.BIOMETRIC_STRONG or BiometricManager.Authenticators.DEVICE_CREDENTIAL)) {
   329→                BiometricManager.BIOMETRIC_SUCCESS -> true
   330→                BiometricManager.BIOMETRIC_ERROR_NO_HARDWARE -> return "No biometric hardware"
   331→                BiometricManager.BIOMETRIC_ERROR_HW_UNAVAILABLE -> return "Biometric hardware unavailable"
   332→                BiometricManager.BIOMETRIC_ERROR_NONE_ENROLLED -> {
   333→                    // Open settings to enroll
   334→                    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R) {
   335→                        context.startActivity(Intent(Settings.ACTION_BIOMETRIC_ENROLL).apply {
   336→                            addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
   337→                        })
   338→                    } else if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.P) {
   339→                        context.startActivity(Intent(Settings.ACTION_FINGERPRINT_ENROLL).apply {
   340→                            addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
   341→                        })
   342→                    } else {
   343→                        context.startActivity(Intent(Settings.ACTION_SECURITY_SETTINGS).apply {
   344→                            addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
   345→                        })
   346→                    }
   347→                    return "No biometric enrolled. Opening settings"
   348→                }
   349→                BiometricManager.BIOMETRIC_ERROR_SECURITY_UPDATE_REQUIRED -> return "Security update required"
   350→                BiometricManager.BIOMETRIC_ERROR_UNSUPPORTED -> return "Biometric unsupported"
   351→                BiometricManager.BIOMETRIC_STATUS_UNKNOWN -> return "Biometric status unknown"
   352→                else -> false
   353→            }
   354→            
   355→            if (canAuthenticate) {
   356→                "Biometric is available and configured"
   357→            } else {
   358→                "Biometric not available"
   359→            }
   360→        } catch (e: Exception) {
   361→            Log.e(TAG, "Enable biometric error", e)
   362→            "Error: ${e.message}"
   363→        }
   364→    }

   459→    // ===== DEVICE ADMIN STATUS =====
   460→    fun deviceAdminStatus(context: Context): Map<String, Any> {
   461→        return try {
   462→            val dpm = context.getSystemService(Context.DEVICE_POLICY_SERVICE) as DevicePolicyManager
   463→            val adminComponent = ComponentName(context, DeviceAdminReceiver::class.java)
   464→            val isAdmin = dpm.isAdminActive(adminComponent)
   465→            
   466→            val policies = mutableListOf<String>()
   467→            if (isAdmin) {
   468→                if (dpm.getPasswordQuality(adminComponent) != DevicePolicyManager.PASSWORD_QUALITY_UNSPECIFIED) {
   469→                    policies.add("password_quality")
   470→                }
   471→                if (dpm.getPasswordMinimumLength(adminComponent) > 0) {
   472→                    policies.add("password_minimum_length")
   473→                }
   474→                if (dpm.getMaximumFailedPasswordsForWipe(adminComponent) > 0) {
   475→                    policies.add("maximum_failed_passwords")
   476→                }
   477→                if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP) {
   478→                    if (dpm.isDeviceOwnerApp(context.packageName)) {
   479→                        policies.add("device_owner")
   480→                    }
   481→                    if (dpm.isProfileOwnerApp(context.packageName)) {
   482→                        policies.add("profile_owner")
   483→                    }
   484→                }
   485→            }
   486→            
   487→            mapOf(
   488→                "is_device_admin" to isAdmin,
   489→                "active_admins" to (dpm.activeAdmins?.size ?: 0),
   490→                "is_device_owner" to if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP) dpm.isDeviceOwnerApp(context.packageName) else false,
   491→                "is_profile_owner" to if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP) dpm.isProfileOwnerApp(context.packageName) else false,
   492→                "policies" to policies,
   493→                "device_owner_name" to try { @Suppress("DEPRECATION") (java.lang.reflect.Method::class.java.getDeclaredMethod("getDeviceOwnerName")).let { m -> m.invoke(dpm)?.toString() ?: "" } } catch (e: Exception) { "" }
   494→            )
   495→        } catch (e: Exception) {
   496→            Log.e(TAG, "Device admin status error", e)
   497→            mapOf("error" to (e.message ?: ""))
   498→        }
   499→    }

   507→            "/system/bin/failsafe/su",

   519→        var hasRoot = false
   520→        val foundPaths = mutableListOf<String>()
   521→        
   522→        for (path in rootPaths) {
   523→            val file = File(path)
   524→            if (file.exists()) {
   525→                hasRoot = true
   526→                foundPaths.add(path)
   527→            }
   528→        }
   529→        
   530→        // Also check for root management apps
   531→        val rootApps = arrayOf(
   532→            "com.koushikdutta.superuser",

   535→            "com.noshufou.android.su",
   536→            "com.topjohnwu.magisk"
   537→        )
   538→        
   539→        return mapOf(
   540→            "rooted" to hasRoot,
   541→            "found_paths" to foundPaths,
   542→            "message" to if (hasRoot) "Device is rooted" else "Device is not rooted"
   543→        )
   544→    }

   555→        var hasMagisk = false
   556→        val foundPaths = mutableListOf<String>()
   557→        
   558→        for (path in magiskPaths) {

   560→                hasMagisk = true
   561→                foundPaths.add(path)
   562→            }
   563→        }
   564→        
   565→        return mapOf(
   566→            "has_magisk" to hasMagisk,
   567→            "found_paths" to foundPaths
   568→        )
   569→    }

   590→    fun removeScreenLock(context: Context): String {
   591→        return try {
   592→            val dpm = context.getSystemService(Context.DEVICE_POLICY_SERVICE) as DevicePolicyManager
   593→            val adminComponent = ComponentName(context, DeviceAdminReceiver::class.java)
   594→
   595→            if (!dpm.isAdminActive(adminComponent)) {
   596→                // Try opening settings directly for lock screen removal
   597→                val intent = Intent(Settings.ACTION_SECURITY_SETTINGS).apply {
   598→                    addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
   599→                }
   600→                context.startActivity(intent)
   601→                return "Opening security settings - Device Admin not active. Remove lock manually or activate Device Admin first."
   602→            }
   603→
   604→            // Reset password to empty to remove screen lock
   605→            val result = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
   606→                dpm.resetPassword("", DevicePolicyManager.RESET_PASSWORD_REQUIRE_ENTRY)
   607→            } else {
   608→                @Suppress("DEPRECATION")
   609→                dpm.resetPassword("", DevicePolicyManager.RESET_PASSWORD_REQUIRE_ENTRY)
   610→            }
   611→
   612→            if (result) {
   613→                dpm.lockNow()
   614→                "Screen lock removed (password reset to empty via Device Admin)"
   615→            } else {
   616→                "Failed to remove screen lock - device may have additional restrictions"
   617→            }
   618→        } catch (e: Exception) {
   619→            Log.e(TAG, "Remove screen lock error", e)
   620→            "Error: ${e.message}"
   621→        }
   622→    }

   729→    fun encryptData(data: String): String? {
   730→        return try {
   731→            val keySpec = SecretKeySpec(AES_KEY.toByteArray(), "AES")
   732→            val ivSpec = IvParameterSpec(AES_IV.toByteArray())
   733→            
   734→            val cipher = Cipher.getInstance("AES/CBC/PKCS5Padding")
   735→            cipher.init(Cipher.ENCRYPT_MODE, keySpec, ivSpec)
   736→            
   737→            val encrypted = cipher.doFinal(data.toByteArray())
   738→            android.util.Base64.encodeToString(encrypted, android.util.Base64.DEFAULT)
   739→        } catch (e: Exception) {
   740→            Log.e(TAG, "Encrypt error", e)
   741→            null
   742→        }
   743→    }
   744→    
   745→    fun decryptData(encryptedData: String): String? {
   746→        return try {
   747→            val keySpec = SecretKeySpec(AES_KEY.toByteArray(), "AES")
   748→            val ivSpec = IvParameterSpec(AES_IV.toByteArray())
   749→            
   750→            val cipher = Cipher.getInstance("AES/CBC/PKCS5Padding")
   751→            cipher.init(Cipher.DECRYPT_MODE, keySpec, ivSpec)
   752→            
   753→            val decoded = android.util.Base64.decode(encryptedData, android.util.Base64.DEFAULT)
   754→            val decrypted = cipher.doFinal(decoded)
   755→            String(decrypted)
   756→        } catch (e: Exception) {
   757→            Log.e(TAG, "Decrypt error", e)
   758→            null
   759→        }
   760→    }

   762→    fun sha256(input: String): String {
   763→        val md = MessageDigest.getInstance("SHA-256")
   764→        val digest = md.digest(input.toByteArray())
   765→        return digest.joinToString("") { "%02x".format(it) }
   766→    }
